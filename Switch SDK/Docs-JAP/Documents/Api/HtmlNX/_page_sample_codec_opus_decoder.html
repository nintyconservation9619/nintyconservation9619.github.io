Opus デコーダのサンプルプログラムの解説です。

概要

Opus デコーダを用いて、Opus データをデコードしながらオーディオ再生を行うサンプルです。 オーディオ再生にはオーディオレンダラを利用します。

ファイル構成

本サンプルプログラムは Samples/Sources/Applications/CodecOpusDecoder 以下にあります。

必要な環境

オーディオ出力が利用可能である必要があります。

操作方法

サンプルプログラムを実行すると、一定の時間 BGM を再生し、自動的に終了します。

注意事項

特にありません。

実行手順

サンプルプログラムをビルドし、実行してください。

151 extern "C" void nnMain()
152 {
153 #ifdef USE_HARDWARE_DECODER
154  NNS_LOG("CodecOpusDecoder sample (using nn::codec::HardwareOpusDecoder)\n");
155 #else // USE_HARDWARE_DECODER
156  NNS_LOG("CodecOpusDecoder sample (using nn::codec::OpusDecoder)\n");
157 #endif // USE_HARDWARE_DECODER
158 
159  nn::mem::StandardAllocator allocator(g_HeapBuffer, sizeof(g_HeapBuffer));
160  nn::mem::StandardAllocator pcmBufferAllocator(g_PcmBuffer, sizeof(g_PcmBuffer));
161 
162  // ファイルシステムを初期化します
163  InitializeFileSystem();
164 
165  // Opus データを読み込みます
166  const char Filename[] = "asset:/SampleBgm0-1ch.opus";
167  nn::fs::FileHandle handle;
168  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::OpenFile(&handle, Filename, nn::fs::OpenMode_Read));
169  int64_t size;
170  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::GetFileSize(&size, handle));
171  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::ReadFile(handle, 0, g_OpusData, static_cast<std::size_t>(size)));
172  nn::fs::CloseFile(handle);
173 
174  // ヘッダより、サンプルレートとチャンネル数を取得します
175  nn::codec::OpusBasicInfo opusBasicInfo;
176  std::memcpy(&opusBasicInfo, g_OpusData, sizeof(opusBasicInfo));
177  NN_ABORT_UNLESS(opusBasicInfo.header.type == nn::codec::OpusInfoType_BasicInfo);
178  int sampleRate = opusBasicInfo.sampleRate;
179  int channelCount = opusBasicInfo.channelCount;
180 
181  // Opus デコーダを初期化します
182 #ifndef USE_HARDWARE_DECODER
183  nn::codec::OpusDecoder decoder;
184 #else // #ifndef USE_HARDWARE_DECODER
185  const int32_t noOption = 0; // 現在、指定可能なオプションはありません
186  nn::codec::HardwareOpusDecoder decoder(noOption);
187 #endif // #ifndef USE_HARDWARE_DECODER
188  void* workBuffer = std::malloc(decoder.GetWorkBufferSize(sampleRate, channelCount));
189  NN_ABORT_UNLESS_NOT_NULL(workBuffer);
190  nn::codec::OpusResult opusResult = decoder.Initialize(sampleRate, channelCount, workBuffer, decoder.GetWorkBufferSize(sampleRate, channelCount));
191  NN_ABORT_UNLESS(opusResult == nn::codec::OpusResult_Success);
192 
193  //オーディオレンダラを構築します
194  nn::audio::AudioRendererParameter audioRendererParameter;
195  nn::audio::InitializeAudioRendererParameter(&audioRendererParameter);
196  audioRendererParameter.sampleRate = 48000;
197  audioRendererParameter.sampleCount = 240;
198  audioRendererParameter.mixBufferCount = 2;
199  audioRendererParameter.voiceCount = channelCount;
200  audioRendererParameter.sinkCount = 1;
201  NN_ABORT_UNLESS(nn::audio::IsValidAudioRendererParameter(audioRendererParameter));
202 
203  std::size_t audioRendererWorkBufferSize = nn::audio::GetAudioRendererWorkBufferSize(audioRendererParameter);
204  void* audioRendererWorkBuffer = allocator.Allocate(audioRendererWorkBufferSize, nn::os::MemoryPageSize);
205  NN_ABORT_UNLESS_NOT_NULL(audioRendererWorkBuffer);
206 
207  nn::audio::AudioRendererHandle audioRendererHandle;
208  nn::os::SystemEvent systemEvent;
209  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::audio::OpenAudioRenderer(&audioRendererHandle, &systemEvent, audioRendererParameter, audioRendererWorkBuffer, audioRendererWorkBufferSize));
210 
211  std::size_t configBufferSize = nn::audio::GetAudioRendererConfigWorkBufferSize(audioRendererParameter);
212  void* configBuffer = allocator.Allocate(configBufferSize, nn::os::MemoryPageSize);
213  NN_ABORT_UNLESS_NOT_NULL(configBuffer);
214  nn::audio::AudioRendererConfig config;
215  nn::audio::InitializeAudioRendererConfig(&config, audioRendererParameter, configBuffer, configBufferSize);
216 
217  nn::audio::FinalMixType finalMix;
218  nn::audio::AcquireFinalMix(&config, &finalMix, 2);
219 
220  nn::audio::DeviceSinkType sink;
221  int8_t bus[2] = { 0, 1 };
222  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::audio::AddDeviceSink(&config, &sink, &finalMix, bus, sizeof(bus) / sizeof(bus[0]), "MainAudioOut"));
223 
224  // WaveBuffer に追加するオーディオデータを保持するためのメモリプールを準備します。
225  nn::audio::MemoryPoolType pcmBufferMemoryPool;
226  NN_ABORT_UNLESS(nn::audio::AcquireMemoryPool(&config, &pcmBufferMemoryPool, g_PcmBuffer, sizeof(g_PcmBuffer)));
227  NN_ABORT_UNLESS(nn::audio::RequestAttachMemoryPool(&pcmBufferMemoryPool));
228 
229  nn::audio::VoiceType voice;
230  nn::audio::AcquireVoiceSlot(&config, &voice, sampleRate, channelCount, nn::audio::SampleFormat_PcmInt16, nn::audio::VoiceType::PriorityHighest, nullptr, 0);
231  nn::audio::SetVoiceDestination(&config, &voice, &finalMix);
232  nn::audio::SetVoicePlayState(&voice, nn::audio::VoiceType::PlayState_Play);
233  if(channelCount == 1)
234  {
235  nn::audio::SetVoiceMixVolume(&voice, &finalMix, 1.0f, 0, 0);
236  nn::audio::SetVoiceMixVolume(&voice, &finalMix, 1.0f, 0, 1);
237  }
238  else
239  {
240  nn::audio::SetVoiceMixVolume(&voice, &finalMix, 1.0f, 0, 0);
241  nn::audio::SetVoiceMixVolume(&voice, &finalMix, 1.0f, 1, 1);
242  }
243 
244  // ストリーム再生用のバッファを用意します
245  // 最初は 0 で初期化したものを渡し、この後のメインループで Opus データのデコード結果を与えています
246  nn::audio::WaveBuffer waveBuffer[BufferCount];
247  void* pcmBuffer[BufferCount];
248  for (int i = 0; i < BufferCount; ++i)
249  {
250  const std::size_t pcmBufferSize = OpusSampleCount * OpusChannelCount * sizeof(int16_t);
251  pcmBuffer[i] = pcmBufferAllocator.Allocate(pcmBufferSize, nn::audio::BufferAlignSize);
252 
253  std::memset(pcmBuffer[i], 0, pcmBufferSize);
254 
255  waveBuffer[i].buffer = pcmBuffer[i];
256  waveBuffer[i].size = pcmBufferSize;
257  waveBuffer[i].startSampleOffset = 0;
258  waveBuffer[i].endSampleOffset = static_cast<int32_t>(pcmBufferSize / sizeof(int16_t));
259  waveBuffer[i].loop = false;
260  waveBuffer[i].isEndOfStream = false;
261  waveBuffer[i].pContext = nullptr;
262  waveBuffer[i].contextSize = 0;
263 
264  nn::audio::AppendWaveBuffer(&voice, &waveBuffer[i]);
265  }
266 
267  // オーディオレンダラを開始します
268  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::audio::StartAudioRenderer(audioRendererHandle));
269 
270  // Opus データ本体と、そのサイズを取得します
271  unsigned char* p = reinterpret_cast<unsigned char*>(g_OpusData + opusBasicInfo.dataInfoOffset);
272  nn::codec::OpusInfoHeader dataInfoHeader;
273  std::memcpy(&dataInfoHeader, p, sizeof(dataInfoHeader));
274  NN_ABORT_UNLESS(dataInfoHeader.type == nn::codec::OpusInfoType_DataInfo);
275  std::size_t remainedCount = dataInfoHeader.size;
276  p += sizeof(dataInfoHeader);
277 
278  int bufferCount = BufferCount;
279  for (;;)
280  {
281  systemEvent.Wait();
282  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::audio::RequestUpdateAudioRenderer(audioRendererHandle, &config));
283 
284  // 再生が完了したバッファがあれば、Opus デコードを実行します
285  if (auto pReleasedWaveBuffer = nn::audio::GetReleasedWaveBuffer(&voice))
286  {
287  nn::audio::WaveBuffer* pWaveBuffer = nullptr;
288  int16_t* pPcmBuffer = nullptr;
289  for (int i = 0; i < BufferCount; ++i)
290  {
291  if (pReleasedWaveBuffer == &waveBuffer[i])
292  {
293  pWaveBuffer = &waveBuffer[i];
294  pPcmBuffer = static_cast<int16_t*>(pcmBuffer[i]);
295  break;
296  }
297  }
298 
299  // データが残っていればデコードを実行します
300  if (remainedCount > 0)
301  {
302  std::size_t consumed;
303  int sampleCount;
304  opusResult = decoder.DecodeInterleaved(&consumed, &sampleCount, pPcmBuffer, OpusSampleCount * OpusChannelCount * sizeof(int16_t), p, remainedCount);
305  NN_ABORT_UNLESS(opusResult == nn::codec::OpusResult_Success);
306  p += consumed;
307  remainedCount -= consumed;
308 
309  pWaveBuffer->size = sampleCount * sizeof(int16_t) * channelCount;
310  pWaveBuffer->endSampleOffset = sampleCount;
311  pWaveBuffer->loop = false;
312  pWaveBuffer->isEndOfStream = false;
313 
314  nn::audio::AppendWaveBuffer(&voice, pWaveBuffer);
315  }
316  // データが残っていなければ、全てのバッファの再生が完了するまで待ちます
317  else if (--bufferCount == 0)
318  {
319  break;
320  }
321  }
322  }
323 
324  // 利用していたメモリプールにデタッチ要求をします。
325  nn::audio::RequestDetachMemoryPool(&pcmBufferMemoryPool);
326  // メモリプールがデタッチされるのを待ちます。
327  while (nn::audio::IsMemoryPoolAttached(&pcmBufferMemoryPool))
328  {
329  nn::audio::GetReleasedWaveBuffer(&voice);
330 
331  systemEvent.Wait();
332  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::audio::RequestUpdateAudioRenderer(audioRendererHandle, &config));
333  }
334  // MemoryPool を解放する。
335  nn::audio::ReleaseMemoryPool(&config, &pcmBufferMemoryPool);
336  // バッファを解放する。
337  for (int i = 0; i < BufferCount; ++i)
338  {
339  pcmBufferAllocator.Free(pcmBuffer[i]);
340  }
341 
342  // オーディオレンダラを終了し、メモリを破棄します
343  nn::audio::StopAudioRenderer(audioRendererHandle);
344  nn::audio::CloseAudioRenderer(audioRendererHandle);
345  nn::os::DestroySystemEvent(systemEvent.GetBase());
346  allocator.Free(audioRendererWorkBuffer);
347  allocator.Free(configBuffer);
348 
349  // Opus デコーダを終了し、メモリを破棄します
350  decoder.Finalize();
351  std::free(workBuffer);
352 
353  // ファイルシステムの終了処理を行います
354  FinalizeFileSystem();
355 
356 } // NOLINT(readability/fn_size)

サンプルプログラムの解説

このサンプルプログラムは、あらかじめエンコードされた Opus データをデコードしながら、オーディオ再生を行うものです。

サンプルプログラムの処理の流れは以下の通りです。

  • ファイルシステムの初期化
  • Opus データの読み込み
  • Opus デコーダの初期化
  • オーディオレンダラの初期化、および開始
  • Opus デコーダの実行と、デコード結果の再生
  • オーディオレンダラの終了
  • Opus デコーダの終了

サンプルプログラムでは、デフォルトで Opus デコーダに nn::codec::OpusDecoder が使用されます。 USE_HARDWARE_DECODER マクロを定義してビルドすると nn::codec::HardwareOpusDecoder が使用されます。 以下の説明は nn::codec::OpusDecoder を代表例としています。 nn::codec::HardwareOpusDecoder を使用する場合は、OpusDecoder を HardwareOpusDecoder に置き換えてお読み下さい。

最初にファイルシステムの初期化を行います。 リソースファイルはビルド時にコンバートされるものを利用します。 リソースディレクトリの場所は適宜変更して利用してください。

Opus データをファイルから読み込みます。

ヘッダ情報をもとに Opus デコーダを初期化します。

初期化関数 nn::codec::OpusDecoder::Initialize() にはサンプルレートとチャンネル数、さらにそれらによってサイズが決まるワークバッファを与える必要があります。

ワークバッファのサイズは nn::codec::OpusDecoder::GetWorkBufferSize() により取得可能です。

オーディオレンダラを準備します。 オーディオレンダラの詳細に関しては Samples/Sources/Applications/AudioRenderer サンプルおよび関連する API リファレンスを参照してください。

メインループでは、ストリーム再生のような処理を行います。 複数の PCM バッファを用意し、 nn::audio::VoiceType 経由で再生しながら、 再生が終わったバッファに対して順次 Opus データのデコードを行います。

Opus データをすべてデコードし、再生を終えると、オーディオレンダラを停止し、 オーディオレンダラおよび Opus デコーダを終了し、メモリの破棄を行います。
