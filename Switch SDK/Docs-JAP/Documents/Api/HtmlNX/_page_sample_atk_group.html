nn::atk ライブラリのグループを使ったサウンドデータの読み込みに関する機能のサンプルプログラムの解説です。

概要

Atk ライブラリでグループを使ったサウンドデータの読み込みを行うためのサンプルです。

ファイル構成

本サンプルプログラムは Samples/Sources/Applications/AtkGroup 以下にあります。

必要な環境

とくになし

操作方法

コンソールに操作方法が表示されます。

注意事項

実行中のログが、画面上とコンソールに表示され、 起動直後のログとして操作方法が表示されます。

PC 版ではウィンドウは立ち上がらず、コンソール表示のみになる点にご注意ください。

実行手順

サンプルプログラムをビルドし、実行してください。

151 void InitializeAtk()
152 {
153  bool isSuccess = true;
154 
155  // SoundSystem の初期化
156  nn::atk::SoundSystem::SoundSystemParam param;
157  std::size_t memSizeForSoundSystem = nn::atk::SoundSystem::GetRequiredMemSize( param );
158  g_pMemoryForSoundSystem = nns::atk::Allocate( memSizeForSoundSystem, nn::atk::SoundSystem::WorkMemoryAlignSize );
159  isSuccess = nn::atk::SoundSystem::Initialize(
160  param,
161  reinterpret_cast<uintptr_t>( g_pMemoryForSoundSystem ),
162  memSizeForSoundSystem );
163  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundSystem" );
164 
165  // SoundHeap の初期化
166  g_pMemoryForSoundHeap = nns::atk::Allocate( SoundHeapSize );
167  isSuccess = g_SoundHeap.Create( g_pMemoryForSoundHeap, SoundHeapSize );
168  NN_ABORT_UNLESS( isSuccess, "cannot create SoundHeap" );
169 
170  // SoundArchive の初期化
171  const char* archiveAbsolutePath = nns::atk::GetAbsolutePath(ArchiveRelativePath);
172  isSuccess = g_SoundArchive.Open(archiveAbsolutePath);
173  NN_ABORT_UNLESS( isSuccess, "cannot open SoundArchive(%s)\n", archiveAbsolutePath );
174 
175  // SoundArchive のパラメータ情報をメモリにロード
176  std::size_t infoBlockSize = g_SoundArchive.GetHeaderSize();
177  g_pMemoryForInfoBlock = nns::atk::Allocate( infoBlockSize, nn::atk::FsSoundArchive::BufferAlignSize );
178  isSuccess = g_SoundArchive.LoadHeader( g_pMemoryForInfoBlock, infoBlockSize );
179  NN_ABORT_UNLESS( isSuccess, "cannot load InfoBlock" );
180 
181  // SoundDataManager の初期化
182  std::size_t memSizeForSoundDataManager = g_SoundDataManager.GetRequiredMemSize( &g_SoundArchive );
183  g_pMemoryForSoundDataManager = nns::atk::Allocate( memSizeForSoundDataManager, nn::atk::SoundDataManager::BufferAlignSize );
184  isSuccess = g_SoundDataManager.Initialize(
185  &g_SoundArchive,
186  g_pMemoryForSoundDataManager,
187  memSizeForSoundDataManager );
188  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundDataManager" );
189 
190  // SoundArchivePlayer で用いるストリームバッファの初期化
191  // ストリームバッファはメモリプール管理されているヒープから確保する必要があります。
192  std::size_t memSizeForStreamBuffer = g_SoundArchivePlayer.GetRequiredStreamBufferSize( &g_SoundArchive );
193  g_pMemoryForStreamBuffer = nns::atk::AllocateForMemoryPool(memSizeForStreamBuffer);
194 
195  // 専用のヒープをメモリプールにアタッチ
196  nn::atk::SoundSystem::AttachMemoryPool(&g_MemoryPool, nns::atk::GetPoolHeapAddress(), nns::atk::GetPoolHeapSize());

198  // SoundArchivePlayer の初期化
199  std::size_t memSizeForSoundArchivePlayer = g_SoundArchivePlayer.GetRequiredMemSize( &g_SoundArchive );

234 void LoadData()
235 {
236  bool isSuccess = true;
237 
238  // 「埋め込み」グループのデータの読み込み
239  isSuccess = g_SoundDataManager.LoadData( GROUP_EMBEDDED, &g_SoundHeap );
240  NN_ABORT_UNLESS( isSuccess, "LoadData(GROUP_EMBEDDED) failed." );
241 
242  // 「リンク」グループのデータの読み込み
243  isSuccess = g_SoundDataManager.LoadData( GROUP_LINKED, &g_SoundHeap );
244  NN_ABORT_UNLESS( isSuccess, "LoadData(GROUP_LINKED) failed." );
245 
246  // 「ユーザー管理」グループのグループファイルの読み込み
247  nn::fs::FileHandle fileHandle;
248  for ( int i = 0; i < GroupFileCount; i++ )
249  {
250  const char* GroupFileAbsolutePath = nns::atk::GetAbsolutePath(GroupFilesRelativePath[i]);
251 
252  nn::Result result = nn::fs::OpenFile(&fileHandle, GroupFileAbsolutePath, nn::fs::OpenMode_Read);
253  NN_ABORT_UNLESS( result.IsSuccess() );
254  result = nn::fs::GetFileSize(&g_GroupFileSizes[i], fileHandle);
255  NN_ABORT_UNLESS( result.IsSuccess() );
256  g_pMemoryForGroupFiles[i] = nns::atk::AllocateForMemoryPool( static_cast<size_t>(g_GroupFileSizes[i]) );
257 
258  size_t readSize = 0;
259  result = nn::fs::ReadFile(&readSize, fileHandle, 0, g_pMemoryForGroupFiles[i], static_cast<size_t>(g_GroupFileSizes[i]));
260 
261  NN_ABORT_UNLESS( result.IsSuccess() );
262  nn::fs::CloseFile(fileHandle);
263  }
264  g_IsSetGroupFile = false;
265 }

290 bool UpdateAtk()
291 {
292  // StartSound / Stop
293  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::A >() )
294  {
295  PlayWithStartSound(SE_YOSHI_FOR_LINKEDGROUP, "SE_YOSHI_FOR_LINKEDGROUP");

297  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::X >() )
298  {
299  // SoundDataManager にユーザー管理グループがセットされていない場合は再生に失敗します
300  PlayWithStartSound(SE_YOSHI, "SE_YOSHI");
301  }
302  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Y >() )
303  {
304  PlayWithStartSound(SE_YOSHI_FOR_EMBEDDEDGROUP, "SE_YOSHI_FOR_EMBEDDEDGROUP");
305  }
306  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::B >() )
307  {
308  g_SoundHandle.Stop( 0 );
309  }
310 
311  // Print Usage
312  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::L >() )
313  {
314  PrintUsage();
315  }
316 
317  // Set/Unset Group Files
318  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::R >() )
319  {
320  if ( g_IsSetGroupFile )
321  {
322  // ロード済みグループファイルの登録解除
323  for ( int i = 0; i < GroupFileCount; i++ )
324  {
325  g_SoundDataManager.ClearFileAddressInGroupFile(g_pMemoryForGroupFiles[i], static_cast<size_t>(g_GroupFileSizes[i]));
326  }
327  NNS_LOG("Unset GroupFile Address\n");
328  }
329  else
330  {
331  // ロード済みグループファイルの登録
332  for ( int i = 0; i < GroupFileCount; i++ )
333  {
334  g_SoundDataManager.SetFileAddressInGroupFile(g_pMemoryForGroupFiles[i], static_cast<size_t>(g_GroupFileSizes[i]));
335  }
336  NNS_LOG("Set GroupFile Address\n");
337  }
338  g_IsSetGroupFile = !g_IsSetGroupFile;
339  }
340 
341  // Exit
342  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Start >() )
343  {
344  return false;
345  }
346 
347  g_SoundArchivePlayer.Update();
348 
349  return true;
350 }

サンプルプログラムの解説

先のサンプルプログラムの Atk に関連している処理の流れは以下の通りです(AtkSimple と違う部分には追加要素と書かれています)。

  • アロケータの初期化
    • 内部で２種類（メモリプール管理対象用とそれ以外）のアロケータを初期化。
  • Atk ライブラリの初期化 (InitializeAtk())
    • SoundSystem の初期化
    • SoundHeap の初期化
    • SoundArchive をオープンし、パラメータ情報をロード
    • SoundDataManager の初期化
    • SoundArchivePlayer で用いるストリームバッファをメモリプール管理対象用アロケータから確保し初期化
    • メモリプール管理対象のメモリをメモリプールにアタッチ
    • SoundArchivePlayer の初期化
  • SoundHeap に埋め込みグループとリンクグループをロード (LoadData()) (追加要素)
  • メモリ上にユーザー管理グループをロード (LoadData()) (追加要素)
  • メインループでの処理
    • キー操作があれば再生・停止処理を実行
    • メモリ上にロードしたユーザー管理グループを SoundDataManager に設定・解除 (追加要素)
    • SoundArchivePlayer の更新
  • Atk ライブラリの終了処理 (FinalizeAtk())
    • SoundArchivePlayer の 終了処理
    • メモリプール管理対象のメモリをメモリプールからデタッチ
    • その他の終了処理・メモリ解放
  • アロケータの終了処理

キー操作を行うことで、ウェーブサウンド、シーケンスサウンドを再生します。 このサンプルプログラムでは再生用のAPIとして、StartSound を使用しています。 StartSoundは一度実行すると停止処理を行うか、サウンドの再生が終わるまで再生を続けます。

メモリプール管理対象のメモリのアタッチ・デタッチに関しては以下の点にご注意ください。

  • メモリプールのアタッチ・デタッチは即座に反映されないため、 SoundSystem が初期化された状態で、 アタッチ・デタッチのリクエストを送り、しばらく待つ必要がある
  • 再生中にはメモリプールをデタッチできないため、明示的に停止させるか SoundArchivePlayer::Finalize() を実行したあとにメモリプールをデタッチする必要がある
