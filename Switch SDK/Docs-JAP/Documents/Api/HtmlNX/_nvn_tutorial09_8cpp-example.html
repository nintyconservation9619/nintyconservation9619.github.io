/*
* ShaderSpecialization::Init
* ---------------------------
* Initialize NVN, load asset files, and create objects needed for the
* application to run.
*/
void ShaderSpecialization::Init(PFNNVNBOOTSTRAPLOADERPROC pLoader, NVNnativeWindow nativeWindow)
{
NN_ASSERT(pLoader != NULL, "Bootstrap loader function pointer is NULL\n");
/* Load the function pointer to get other nvn function pointers. */
pfnc_nvnDeviceGetProcAddress = (PFNNVNDEVICEGETPROCADDRESSPROC) (*pLoader)("nvnDeviceGetProcAddress");
if (pfnc_nvnDeviceGetProcAddress == NULL)
{
/* This can happen if an NVN driver is not installed on a Windows PC. */
NN_ASSERT(0, "BootstrapLoader failed to find nvnDeviceGetProcAddress");
}
/*
* Load function pointers for creating nvn device
*/
nvnLoadCProcs( NULL, pfnc_nvnDeviceGetProcAddress );
/* Check the API version. */
int majorVersion, minorVersion;
nvnDeviceGetInteger( NULL, NVN_DEVICE_INFO_API_MAJOR_VERSION, &majorVersion );
nvnDeviceGetInteger( NULL, NVN_DEVICE_INFO_API_MINOR_VERSION, &minorVersion );
if( majorVersion != NVN_API_MAJOR_VERSION || minorVersion < NVN_API_MINOR_VERSION )
{
NN_ASSERT( 0, "NVN SDK not supported by current driver." );
}
/* If debug or develop is enabled, turn on NVN's debug layer. */
int deviceFlags = 0;
#if defined(NN_SDK_BUILD_DEBUG) || defined(NN_SDK_BUILD_DEVELOP)
deviceFlags = NVN_DEVICE_FLAG_DEBUG_ENABLE_LEVEL_2_BIT;
#endif
NVNdeviceBuilder deviceBuilder;
nvnDeviceBuilderSetDefaults(&deviceBuilder);
nvnDeviceBuilderSetFlags(&deviceBuilder, deviceFlags);
if (nvnDeviceInitialize(&m_Device, &deviceBuilder) == false)
{
/*
* This can fail for a few reasons; the most likely on Horizon is
* insufficent device memory.
*/
NN_ASSERT(0, "nvnDeviceInitialize");
}
/*
* Ensure function pointers are all loaded properly
*/
nvnLoadCProcs(&m_Device, pfnc_nvnDeviceGetProcAddress );
/* Setup the debug callback for the debug layer. */
if (deviceFlags & NVN_DEVICE_FLAG_DEBUG_ENABLE_LEVEL_2_BIT)
{
nvnDeviceInstallDebugCallback(
&m_Device,
reinterpret_cast<PFNNVNDEBUGCALLBACKPROC>(&DebugLayerCallback),
NULL, // For testing purposes; any pointer is OK here.
NVN_TRUE // NVN_TRUE = Enable the callback.
);
}
/* Initialize the queue. */
NVNqueueBuilder queueBuilder;
nvnQueueBuilderSetDevice(&queueBuilder, &m_Device);
nvnQueueBuilderSetDefaults(&queueBuilder);
nvnQueueBuilderSetComputeMemorySize(&queueBuilder, 0);
int minQueueCommandMemorySize = 0;
nvnDeviceGetInteger(&m_Device, NVN_DEVICE_INFO_QUEUE_COMMAND_MEMORY_MIN_SIZE, &minQueueCommandMemorySize);
nvnQueueBuilderSetCommandMemorySize(&queueBuilder, minQueueCommandMemorySize);
nvnQueueBuilderSetCommandFlushThreshold(&queueBuilder, minQueueCommandMemorySize);
size_t neededQueueMemorySize = nvnQueueBuilderGetQueueMemorySize(&queueBuilder);
if ((neededQueueMemorySize % NVN_MEMORY_POOL_STORAGE_GRANULARITY) != 0)
{
NN_ASSERT(0, "Memory size reported for queue is not the proper granularity");
}
#if defined( NN_BUILD_TARGET_PLATFORM_OS_WIN )
m_pQueueMemory = NULL;
#else
m_pQueueMemory = AlignedAllocate(neededQueueMemorySize, NVN_MEMORY_POOL_STORAGE_ALIGNMENT);
#endif
nvnQueueBuilderSetQueueMemory(&queueBuilder, m_pQueueMemory, neededQueueMemorySize);
if (nvnQueueInitialize(&m_Queue, &queueBuilder) == false)
{
NN_ASSERT(0, "nvnQueueInitialize failed");
}
/* Create a managed Command Buffer */
m_pRenderTargetManagedCommandBuffer = new ManagedCommandBuffer(&m_Device, g_CommandMemoryChunkSize, g_ControlMemoryChunkSize, g_NumChunks);
/* Builders */
nvnSamplerBuilderSetDevice(&m_SamplerBuilder, &m_Device);
nvnSamplerBuilderSetDefaults(&m_SamplerBuilder);
/* Set up the texture builder for the render target. */
nvnTextureBuilderSetDevice(&m_RenderTargetBuilder, &m_Device);
nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_DISPLAY_BIT | NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, 1920, 1080);
nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_RGBA8);
m_ColorTargetSize = nvnTextureBuilderGetStorageSize(&m_RenderTargetBuilder);
nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, 1920, 1080);
nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_DEPTH32F);
m_DepthTargetSize = nvnTextureBuilderGetStorageSize(&m_RenderTargetBuilder);
/* Allocate the render target memory. */
m_pRenderTargetMemoryPool = new MemoryPool();
m_pRenderTargetMemoryPool->Init(
NULL,
m_ColorTargetSize * g_NumColorBuffers + m_DepthTargetSize,
NVN_MEMORY_POOL_FLAGS_CPU_NO_ACCESS_BIT | NVN_MEMORY_POOL_FLAGS_GPU_CACHED_BIT | NVN_MEMORY_POOL_FLAGS_COMPRESSIBLE_BIT,
&m_Device);
nvnTextureBuilderSetDevice(&m_RenderTargetBuilder, &m_Device);
nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
/* Create the texture ID manager. */
m_pTextureIDManager = new TextureIDManager(&m_Device);
/* Create the asset loader. */
m_pAssetLoader = new AssetFileLoadingHelper(&m_Device, m_pTextureIDManager);
size_t cacheSize = 0;
nn::Result result = nn::fs::QueryMountRomCacheSize(&cacheSize);
NN_ASSERT(result.IsSuccess());
char* mountRomCacheBuffer = new(std::nothrow) char[cacheSize];
NN_ASSERT_NOT_NULL(mountRomCacheBuffer);
result = nn::fs::MountRom("rom", mountRomCacheBuffer, cacheSize);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
if (!nvnSyncInitialize(&m_WindowSync, &m_Device))
{
NN_ASSERT(0, "Failed to initialize window sync");
}
/*
* Load assets in the usual manner using a prepackaged
* asset file. In this tutorial, only textures and
* model data are loaded from the asset file.
* NOTE: Loading specialized shaders is handled
* through a separate function detailed below
*/
m_pDataHolder = m_pAssetLoader->LoadAssetFile("graphicsObjectSpecializedCube.out");
/*
* Open the vertex shader file and load it into a buffer
* that will be passed to the loader as raw source to be
* compiled at runtime
*/
char* ppShaderSources[2];
nn::fs::FileHandle vertexShaderFileHandle;
result = nn::fs::OpenFile(&vertexShaderFileHandle, "rom:/graphicsObjectSpecializedCube.vs", nn::fs::OpenMode_Read);
NN_ASSERT(result.IsSuccess());
int64_t vertexFileSize = 0;
result = nn::fs::GetFileSize(&vertexFileSize, vertexShaderFileHandle);
NN_ASSERT(result.IsSuccess());
ppShaderSources[0] = reinterpret_cast<char*>(malloc(static_cast<size_t>(vertexFileSize + 1)));
memset(ppShaderSources[0], 0, static_cast<size_t>(vertexFileSize + 1));
size_t vertexReadBytes = 0;
result = nn::fs::ReadFile(&vertexReadBytes, vertexShaderFileHandle, 0, ppShaderSources[0], static_cast<size_t>(vertexFileSize));
NN_ASSERT(result.IsSuccess());
/*
* Open the fragment shader file and load it into a buffer
* that will be passed to the loader as raw source to be
* compiled at runtime
*/
nn::fs::FileHandle fragmentShaderFileHandle;
result = nn::fs::OpenFile(&fragmentShaderFileHandle, "rom:/graphicsObjectSpecializedCube.fs", nn::fs::OpenMode_Read);
NN_ASSERT(result.IsSuccess());
int64_t fragmentFileSize;
result = nn::fs::GetFileSize(&fragmentFileSize, fragmentShaderFileHandle);
NN_ASSERT(result.IsSuccess());
ppShaderSources[1] = reinterpret_cast<char*>(malloc(static_cast<size_t>(fragmentFileSize + 1)));
memset(ppShaderSources[1], 0, static_cast<size_t>(fragmentFileSize + 1));
size_t fragmentReadBytes;
result = nn::fs::ReadFile(&fragmentReadBytes, fragmentShaderFileHandle, 0, ppShaderSources[1], static_cast<size_t>(fragmentFileSize));
NN_ASSERT(result.IsSuccess());
/*
* Create a vector to hold the shader stages that
* are referenced in the program that we are
* attempting to specialize.
*/
std::vector<NVNshaderStage> shaderStages;
shaderStages.push_back(NVN_SHADER_STAGE_VERTEX);
shaderStages.push_back(NVN_SHADER_STAGE_FRAGMENT);
/*
* Create a vector to hold the individual specializations of the
* texture style uniform variable in the fragment shader.
* The info for each specialization is stored in a
* GLSLCspecializationUniform object.
*/
std::vector<GLSLCspecializationUniform> specializationTextureStyleData;
const char* textureUniformStyleName = { "u_textureStyle" };
int specializedTexturestyleValues[] = {1, 2, 3, 0};
for(int i = 0; i < 4; ++i)
{
GLSLCspecializationUniform textureStyleInfo;
textureStyleInfo.uniformName = textureUniformStyleName;
textureStyleInfo.values = &specializedTexturestyleValues[i];
textureStyleInfo.numElements = 1;
textureStyleInfo.elementSize = sizeof(int);
specializationTextureStyleData.push_back(textureStyleInfo);
}
/*
* Create a vector to hold the individual specializations of the
* texture index uniform variable in the fragment shader.
*/
std::vector<GLSLCspecializationUniform> specializationTextureIndexData;
const char* textureUniformIndexName = { "u_textureIndex" };
int specializedTextureIndexValues[] = { 0, 1, 2, 3 };
for(int i = 0; i < 4; ++i)
{
GLSLCspecializationUniform textureStyleInfo;
textureStyleInfo.uniformName = textureUniformIndexName;
textureStyleInfo.values = &specializedTextureIndexValues[i];
textureStyleInfo.numElements = 1;
textureStyleInfo.elementSize = sizeof(int);
specializationTextureIndexData.push_back(textureStyleInfo);
}
/*
* A GLSLCspecializationSet object is a collection of all the
* specialized uniform values that will be used for a single
* instance of a shader program.
*
* For this tutorial, a specialization set is generated below
* for each permutation a shader program could have based on
* the 2 uniforms that will be specialized and the 4 values each
* can have, generating 16 unique and separate shader programs
* with these values.
*/
std::vector<GLSLCspecializationSet> specializationSets;
std::vector<std::vector<GLSLCspecializationUniform> > specializationUniforms;
for(int i = 0; i < 4; ++i)
{
for(int j = 0; j < 4; ++j)
{
specializationUniforms.push_back(std::vector<GLSLCspecializationUniform>());
std::vector<GLSLCspecializationUniform>& currentUniformSet = specializationUniforms.back();
currentUniformSet.push_back(specializationTextureStyleData[i]);
currentUniformSet.push_back(specializationTextureIndexData[j]);
GLSLCspecializationSet set;
set.numUniforms = static_cast<uint32_t>(currentUniformSet.size());
set.uniforms = &currentUniformSet[0];
specializationSets.push_back(set);
}
}
/*
* A GLSLCspecializationBatch object is a collection of all the
* specialization sets (generated above) and is passed to the
* specialized compile function.
*
* In summation:
* A Batch contains an array of Sets.
* A Set contains an array of Uniforms.
* A Uniform contains the actual data that a specialized uniform
* variable will have in that particular version of the shader.
*/
GLSLCspecializationBatch specializationBatch;
specializationBatch.numEntries = static_cast<uint32_t>(specializationSets.size());
specializationBatch.entries = &specializationSets[0];
const char* programName = "programName";
/* Localized typedef for convenience */
typedef GraphicsObjectSpecializedCube GOSC;
/*
* Call the function in the asset loader that will compile
* shaders online and generate specializations based on
* the above data.
*/
m_pAssetLoader->LoadSpecializedShaders(
const_cast<const char**>(ppShaderSources), /* Pointer to the array of shader source */
shaderStages, /* Vector listing the shader stages referenced in the shader program */
&specializationBatch, /* Contains the data that is used to specialize the shaders */
programName, /* Name of the shader program */
sizeof(programName), /* Length of shader program's name */
GOSC::ReceiveUniformBlockReflectionInformation, /* Function pointer for grabbing reflection information about uniform blocks*/
GOSC::ReceiveUniformReflectionInformation, /* Function pointer for grabbing reflection information about individual uniforms in blocks*/
ReceiveAttributeReflectionInfo); /* Function pointer for grabbing reflection information about vertex shader attributes */
/*
* Setup vertex attributes with reflection information grabbed with
* ReceiveAttributeReflectionInfo function above
*/
m_pDataHolder->SetupAttributeStatesNVN(GetAttributeLocation);
/*
* Generate simple samplers and bindless texture handles using them
* and textures loaded through the asset loader
*/
m_pDataHolder->SetupTextureSamplerHandle(&m_Device, m_pTextureIDManager, &m_SamplerBuilder);
/* Setup the window builder. */
nvnWindowBuilderSetDefaults(&m_WindowBuilder);
nvnWindowBuilderSetDevice(&m_WindowBuilder, &m_Device);
nvnWindowBuilderSetNativeWindow(&m_WindowBuilder, nativeWindow);
m_pUniformBufferManager = new UniformBufferManager(&m_Device, 64 * 1024);
m_pFrameBufferedSyncManager = new FrameBufferedSyncManager(&m_Device, &m_Queue);
m_pFrameBufferedSyncManager->RegisterMemoryManager(m_pUniformBufferManager);
m_pFrameBufferedSyncManager->RegisterMemoryManager(m_pRenderTargetManagedCommandBuffer);
/*
* Create the render thread pool with four
* worker threads. Each thread will render
* as many of the objects in the work list
* as it can until the list has been exhausted.
*/
m_pRenderThreadPool = new RenderThreadPool();
m_pRenderThreadPool->Init(4, &m_Device, &m_Queue);
/* Create the objects to be rendered by the thread pool. */
for(int i = 0; i < 4; ++i)
{
for (int j = 0; j < 4; ++j)
{
void* place = AlignedAllocate(sizeof(GOSC), NN_ALIGNOF(GOSC));
GOSC* obj = new(place) GOSC(m_pTextureIDManager);
obj->Init(m_pDataHolder);
obj->SetScale(0.5f, 0.5f, 0.5f);
obj->SetTranslate(-1.5f + j, -1.5f + i, -4.0f);
obj->SetRotate(0.0f, 0.0f, 0.0f);
obj->SetupUniforms(m_pUniformBufferManager);
obj->SetShaderIndex((i * 4 + j) % 16);
m_Objects.push_back(obj);
}
}
m_pDebugTextRenderer = DebugTextRenderer::GetInstance();
m_pDebugTextRenderer->Init(&m_Device, &m_Queue, m_pTextureIDManager, m_pFrameBufferedSyncManager);
m_pDebugTextRenderer->SetDrawEnable(true);
m_pDebugTextRenderer->SetSpacing(false);
m_pDebugTextRenderer->SetGridSize(120.0f, 36.0f);
m_pStatTracker = new StatTracker();
m_pStatTracker->Init(&m_Device, CounterFlags_All_Counters, m_pFrameBufferedSyncManager);
/* Clean-up */
free(ppShaderSources[0]);
free(ppShaderSources[1]);
nn::fs::CloseFile(vertexShaderFileHandle);
nn::fs::CloseFile(fragmentShaderFileHandle);
nn::fs::Unmount("rom");
delete[] mountRomCacheBuffer;
}//NOLINT(impl/function_size)
/*
* ShaderSpecialization::UpdateObjectData
* ---------------------------------------
* Updates the transform data of the objects for a
* given frame.
*/
void ShaderSpecialization::UpdateObjectData(int frame)
{
float currentAngle = sin(frame * g_Pi / 180.0f);
float rotX = currentAngle * 15.0f;
float rotY = static_cast<float>(frame % 360);
float rotZ = 0.0f;
for (size_t i = 0; i < m_Objects.size(); ++i)
{
reinterpret_cast<GraphicsObjectSpecializedCube*>(m_Objects[i])->SetRotate(rotX, rotY, rotZ);
}
}
/*
* ShaderSpecialization::ReceiveAttributeReflectionInfo
* ---------------------------------------
* Static function that receives reflection data about vertex attributes
* that is stored in a static vector to be referenced later.
*/
void ShaderSpecialization::ReceiveAttributeReflectionInfo(NVNshaderStageBits shaderStage, int32_t bindingLocation, const std::string& attributeName)
{
AttributeReflectionData data;
data.name = attributeName;
data.location = bindingLocation;
data.stagesReferencedIn = shaderStage;
attributeReflectionDataHolder.push_back(data);
}
/*
* ShaderSpecialization::GetAttributeLocation
* ---------------------------------------
* Static function that accesses the vector filled by the function above
* to retrieve binding locations pulled from the reflection data.
*/
int32_t ShaderSpecialization::GetAttributeLocation(const std::string& name)
{
for(size_t i = 0; i < attributeReflectionDataHolder.size(); ++i)
{
const AttributeReflectionData& data = attributeReflectionDataHolder[i];
if(data.name == name)
{
return data.location;
}
}
return -1;
}
/*
* ShaderSpecialization::UpdateRenderTargets
* ------------------------------------------
* Gets the index of the current render target from the NVNwindow
* and records a command buffer that sets it up to be rendered to
* and clears it.
*/
int ShaderSpecialization::UpdateRenderTargets()
{
/* Get next render target to be used */
NVNwindowAcquireTextureResult result = nvnWindowAcquireTexture(m_pWindow, &m_WindowSync, &m_CurrentWindowIndex);
NN_ASSERT(result == NVN_WINDOW_ACQUIRE_TEXTURE_RESULT_SUCCESS);
/* Record the command buffer to set and clear the target. */
m_pRenderTargetManagedCommandBuffer->BeginRecording();
{
NVNcommandBuffer* commandBuffer = m_pRenderTargetManagedCommandBuffer->GetCommandBuffer();
nvnCommandBufferSetRenderTargets(commandBuffer, 1, &m_RenderTargets[m_CurrentWindowIndex], NULL, m_pDepthTarget, NULL);
nvnCommandBufferSetScissor(commandBuffer, 0, 0, m_ScreenWidth, m_ScreenHeight);
nvnCommandBufferSetViewport(commandBuffer, 0, 0, m_ScreenWidth, m_ScreenHeight);
float clear_color[4] = { 0.4f, 0.55f, 0.6f, 1.0f };
nvnCommandBufferClearColor(commandBuffer, 0, clear_color, NVN_CLEAR_COLOR_MASK_RGBA);
nvnCommandBufferClearDepthStencil(commandBuffer, 1.0, NVN_TRUE, 0, 0);
}
m_RenderTargetCommandHandle = m_pRenderTargetManagedCommandBuffer->EndRecording();
/* Return the index. */
return m_CurrentWindowIndex;
}
/*
* ShaderSpecialization::Shutdown
* ---------------------------------
* This method cleans up all nvn objects and dynamically allocated memory.
*/
void ShaderSpecialization::Shutdown()
{
nvnQueueFinish(&m_Queue);
if(m_pStatTracker)
{
m_pStatTracker->Shutdown();
delete m_pStatTracker;
m_pStatTracker = NULL;
}
if (m_pRenderTargetManagedCommandBuffer)
{
delete m_pRenderTargetManagedCommandBuffer;
m_pRenderTargetManagedCommandBuffer = NULL;
}
if (m_pWindow != NULL)
{
nvnWindowFinalize(m_pWindow);
delete m_pWindow;
m_pWindow = NULL;
}
if (m_pDepthTarget != NULL)
{
nvnTextureFinalize(m_pDepthTarget);
delete m_pDepthTarget;
m_pDepthTarget = NULL;
}
for (int i = 0; i < g_NumColorBuffers; ++i)
{
if (m_RenderTargets[i] != NULL)
{
nvnTextureFinalize(m_RenderTargets[i]);
delete m_RenderTargets[i];
m_RenderTargets[i] = NULL;
}
}
if (m_pDataHolder)
{
delete m_pDataHolder;
m_pDataHolder = NULL;
}
if (m_pAssetLoader != NULL)
{
delete m_pAssetLoader;
m_pAssetLoader = NULL;
}
if (m_pTextureIDManager != NULL)
{
delete m_pTextureIDManager;
m_pTextureIDManager = NULL;
}
for (size_t i = 0; i < m_Objects.size(); ++i)
{
AlignedDeallocate(m_Objects[i]);
}
m_Objects.clear();
if(m_pRenderThreadPool)
{
delete m_pRenderThreadPool;
m_pRenderThreadPool = NULL;
}
if (m_pDebugTextRenderer != NULL)
{
m_pDebugTextRenderer->CleanUp();
m_pDebugTextRenderer = NULL;
}
if(m_pRenderTargetMemoryPool)
{
m_pRenderTargetMemoryPool->Shutdown();
delete m_pRenderTargetMemoryPool;
m_pRenderTargetMemoryPool = NULL;
}
if (m_pUniformBufferManager)
{
delete m_pUniformBufferManager;
m_pUniformBufferManager = NULL;
}
if (m_pFrameBufferedSyncManager != NULL)
{
delete m_pFrameBufferedSyncManager;
m_pFrameBufferedSyncManager = NULL;
}
nvnQueueFinalize(&m_Queue);
if (m_pQueueMemory)
{
AlignedDeallocate(m_pQueueMemory);
m_pQueueMemory = NULL;
}
nvnDeviceFinalize(&m_Device);
}
/*
* ShaderSpecialization::Draw
* ---------------------------
* This function runs the threads in the render thread
* pool to generate the command buffers to draw the list
* of objects and submits them to the queue. It also presents
* the current render target to the screen. Additionally,
* this function uses the StatTracker helper class to display
* GPU performance counters to the screen.
*/
void ShaderSpecialization::Draw(uint64_t millisec)
{
NN_UNUSED(millisec);
static int s_Frame = 0;
++s_Frame;
/*
* Reset the counter values. This causes the GPU counters to be reset for the
* current frame. If this is not called, the values will be accumulated until
* reset is called.
*/
m_pStatTracker->ResetCounters(&m_Queue);
/* Get the current render target and setup/submit a command buffer to set it. */
int index = UpdateRenderTargets();
/* Update the transform data for the objects to be drawn. */
UpdateObjectData(s_Frame);
/*
* Run the threads in the thread pool to generate
* command buffers to render the objects in the work
* list.
*/
std::vector<NVNcommandHandle> outHandles;
m_pRenderThreadPool->Run(&outHandles, &m_Objects);
m_pRenderThreadPool->Wait();
m_pRenderThreadPool->SwapCommandMemory();
/* Insert a fence in the sync manager. */
m_pFrameBufferedSyncManager->InsertFence();
nvnSyncWait(&m_WindowSync, NVN_WAIT_TIMEOUT_MAXIMUM);
/* Submit command buffer for switching render targets */
nvnQueueSubmitCommands(&m_Queue, 1, &m_RenderTargetCommandHandle);
/* Submit the command buffers generated by the thread pool. */
nvnQueueSubmitCommands(&m_Queue, static_cast<int>(outHandles.size()), &outHandles[0]);
/* Copy the counters to the CPU. */
m_pStatTracker->GetCounters(&m_Queue);
/*
* Reports the current values of the GPU counters. The placement of reset
* and report in this application is such that only the rendered models get
* counted in the statistics while the text rendering is left out.
*/
m_pStatTracker->ReportCounters(&m_Queue);
/* Print the counters to the screen with the text renderer. */
m_pStatTracker->PrintCounters(m_pDebugTextRenderer, 0, 0);
/* Render the text. */
m_pDebugTextRenderer->Draw(m_RenderTargets[index], m_ScreenWidth, m_ScreenHeight);
/* Present the render target to the screen. */
nvnQueuePresentTexture(&m_Queue, m_pWindow, index);
/* Swap memory for command buffer and uniform buffers */
m_pFrameBufferedSyncManager->SwapPools();
}
/*
* ShaderSpecialization::Resize
* -----------------------------
* This method is called everytime the window is resized and is passed
* the new size of the window. It frees the old render target and creates a new
* render target with the new screen size.
*/
void ShaderSpecialization::Resize(int width, int height)
{
m_ScreenWidth = width;
m_ScreenHeight = height;
/* Check for the window being minimized or having no visible surface. */
if (width == 0 || height == 0)
{
return;
}
/* If it's the first time Resize is called, allocate the NVNwindow. */
if (m_pWindow == NULL)
{
m_pWindow = new NVNwindow;
}
/*
* Otherwise finalize (free) the NVNwindow used for the previous window size.
* The NVNWindow must be finalized before a render target it owns is finalized.
*/
else
{
nvnWindowFinalize(m_pWindow);
}
/* Set up the builder for the render target. */
nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_DISPLAY_BIT | NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_RGBA8);
for(int i = 0; i < g_NumColorBuffers; ++i)
{
/* If it's the first time Resize is called, allocate the render target. */
if(m_RenderTargets[i] == NULL)
{
m_RenderTargets[i] = new NVNtexture;
}
/* Otherwise finalize (free) the render target used for the previous window size. */
else
{
nvnTextureFinalize(m_RenderTargets[i]);
}
nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * i);
/* Create the texture using the current state of the texture builder. */
nvnTextureInitialize(m_RenderTargets[i], &m_RenderTargetBuilder);
}
if (m_pDepthTarget == NULL)
{
m_pDepthTarget = new NVNtexture;
}
else
{
nvnTextureFinalize(m_pDepthTarget);
}
/* Initialize depth buffer for render target. */
nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_DEPTH32F);
nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * g_NumColorBuffers);
nvnTextureInitialize(m_pDepthTarget, &m_RenderTargetBuilder);
/* Pass off the render targets to the window. */
nvnWindowBuilderSetTextures(&m_WindowBuilder, g_NumColorBuffers, m_RenderTargets);
nvnWindowInitialize(m_pWindow, &m_WindowBuilder);
/* Remake the projection matrix with the new screen size. */
nn::util::Matrix4x4fType projMat44;
nn::util::MatrixPerspectiveFieldOfViewRightHanded(&projMat44, 60.0f * 3.14159f / 180.0f, static_cast<float>(m_ScreenWidth) / static_cast<float>(m_ScreenHeight), 1.0f, 1000.0f);
nn::util::Matrix4x4fType camMat44;
nn::util::MatrixIdentity(&camMat44);
for (size_t i = 0; i < m_Objects.size(); ++i)
{
reinterpret_cast<GraphicsObjectSpecializedCube*>(m_Objects[i])->SetProjectionMatrix(projMat44);
reinterpret_cast<GraphicsObjectSpecializedCube*>(m_Objects[i])->SetCameraMatrix(camMat44);
}
}

}
