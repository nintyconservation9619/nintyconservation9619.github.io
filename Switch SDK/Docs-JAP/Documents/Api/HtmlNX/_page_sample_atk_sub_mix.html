nn::atk ライブラリでサブミックス構成をユーザが定義してサウンドを再生するサンプルプログラムの解説です。

概要

nn::atk ライブラリでサブミックス構成をユーザが定義してサウンドを再生するサンプルです。

ファイル構成

本サンプルプログラムは Samples/Sources/Applications/AtkSubMix 以下にあります。

必要な環境

とくになし

操作方法

コンソールに操作方法が表示されます。

注意事項

実行中のログが、画面上とコンソールに表示され、 起動直後のログとして操作方法が表示されます。

PC 版ではウィンドウは立ち上がらず、コンソール表示のみになる点にご注意ください。

実行手順

サンプルプログラムをビルドし、実行してください。

160 void InitializeSubMix()
161 {
162  // SubMix[A] --> SubMix[B] --+
163  // +--> FinalMix
164  // SubMix[C] ----------------+
165  //
166  // SubMix[A] : バス数 2, チャンネル数 6 (2 番目のバスには EffectReverb を追加)
167  // SubMix[B] : バス数 1, チャンネル数 6
168  // SubMix[C] : バス数 1, チャンネル数 6
169  //
170  // というサブミックスを構成します
171 
172  // SubMix[A] の初期化
173  std::size_t memSizeForSubMixA = nn::atk::SubMix::GetRequiredMemorySize(
174  SubMixBusCount[SubMixIndex_A],
175  SubMixChannelCount,
176  SubMixBusCount[SubMixIndex_B],
177  SubMixChannelCount
178  );
179  g_pMemoryForSubMix[SubMixIndex_A] = nns::atk::Allocate( memSizeForSubMixA );
180  g_SubMix[SubMixIndex_A].Initialize(
181  SubMixBusCount[SubMixIndex_A],
182  SubMixChannelCount,
183  SubMixBusCount[SubMixIndex_B],
184  SubMixChannelCount,
185  g_pMemoryForSubMix[SubMixIndex_A],
186  memSizeForSubMixA
187  );
188 
189  // SubMix[B] の初期化
190  nn::atk::FinalMix& finalMix = nn::atk::SoundSystem::GetFinalMix();
191  std::size_t memSizeForSubMixB = nn::atk::SubMix::GetRequiredMemorySize(
192  SubMixBusCount[SubMixIndex_B],
193  SubMixChannelCount,
194  finalMix.GetBusCount(),
195  finalMix.GetChannelCount()
196  );
197  g_pMemoryForSubMix[SubMixIndex_B] = nns::atk::Allocate( memSizeForSubMixB );
198  g_SubMix[SubMixIndex_B].Initialize(
199  SubMixBusCount[SubMixIndex_B],
200  SubMixChannelCount,
201  finalMix.GetBusCount(),
202  finalMix.GetChannelCount(),
203  g_pMemoryForSubMix[SubMixIndex_B],
204  memSizeForSubMixB
205  );
206 
207  // SubMix[C] の初期化
208  std::size_t memSizeForSubMixC = nn::atk::SubMix::GetRequiredMemorySize(
209  SubMixBusCount[SubMixIndex_C],
210  SubMixChannelCount,
211  finalMix.GetBusCount(),
212  finalMix.GetChannelCount()
213  );
214  g_pMemoryForSubMix[SubMixIndex_C] = nns::atk::Allocate( memSizeForSubMixC );
215  g_SubMix[SubMixIndex_C].Initialize(
216  SubMixBusCount[SubMixIndex_C],
217  SubMixChannelCount,
218  finalMix.GetBusCount(),
219  finalMix.GetChannelCount(),
220  g_pMemoryForSubMix[SubMixIndex_C],
221  memSizeForSubMixC
222  );
223 
224  // SubMix のそれぞれを接続
225  g_SubMix[SubMixIndex_A].SetDestination( &g_SubMix[SubMixIndex_B] );
226  g_SubMix[SubMixIndex_B].SetDestination( &finalMix );
227  g_SubMix[SubMixIndex_C].SetDestination( &finalMix );
228 
229  // SubMix の音量設定
230  g_SubMix[SubMixIndex_C].SetSubMixVolume( 0.2f, 0 ); // SubMix[C] から再生されている事が確認できるよう音量を下げます
231  // SubMix[A] のエフェクトバスを SubMix[B] の先頭バスに出力するよう設定します。
232  // その他のバスのセンド量は、同じインデクスのバスについては初期状態でセンド量が設定されているため、設定の必要はありません。
233  g_SubMix[SubMixIndex_A].SetSend( AppendEffectBusIndex, 0, 1.0f );
234 
235  g_TargetSubMixIndex = SubMixIndex_A;
236 }

238 void FinalizeSubMix()
239 {
240  const auto WaitTime = nn::TimeSpan::FromMilliSeconds( 5 );
241 
242  // SubMix は接続元から終了処理する必要があります。
243  // つまり、この例では SubMix[B] よりも先に SubMix[A] を終了処理する必要があります。
244  while( !g_SubMix[SubMixIndex_A].IsFinalizable() )
245  {
246  nn::os::SleepThread( WaitTime );
247  }
248  g_SubMix[SubMixIndex_A].Finalize();
249 
250  while( !g_SubMix[SubMixIndex_B].IsFinalizable() )
251  {
252  nn::os::SleepThread( WaitTime );
253  }
254  g_SubMix[SubMixIndex_B].Finalize();
255 
256  while( !g_SubMix[SubMixIndex_C].IsFinalizable() )
257  {
258  nn::os::SleepThread( WaitTime );
259  }
260  g_SubMix[SubMixIndex_C].Finalize();
261 }

263 void InitializeAtk()
264 {
265  bool isSuccess = true;
266 
267  // SubMix を使用するための設定
268  int totalBusCount = 0;
269  for(int i = 0; i < SubMixIndex_Count; i++)
270  {
271  totalBusCount += SubMixBusCount[i];
272  }
273 
274  nn::atk::SoundSystem::SoundSystemParam param;
275  param.enableCustomSubMix = true;
276  param.subMixTotalChannelCount = totalBusCount * SubMixChannelCount;
277  param.subMixCount = SubMixIndex_Count;
278 
279  // SoundSystem の初期化
280  std::size_t memSizeForSoundSystem = nn::atk::SoundSystem::GetRequiredMemSize( param );
281  g_pMemoryForSoundSystem = nns::atk::Allocate( memSizeForSoundSystem, nn::atk::SoundSystem::WorkMemoryAlignSize );
282  isSuccess = nn::atk::SoundSystem::Initialize(
283  param,
284  reinterpret_cast<uintptr_t>( g_pMemoryForSoundSystem ),
285  memSizeForSoundSystem );
286  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundSystem" );
287 
288  // SubMix の初期化
289  InitializeSubMix();
290 
291  // SoundHeap の初期化
292  g_pMemoryForSoundHeap = nns::atk::Allocate( SoundHeapSize );
293  isSuccess = g_SoundHeap.Create( g_pMemoryForSoundHeap, SoundHeapSize );
294  NN_ABORT_UNLESS( isSuccess, "cannot create SoundHeap" );
295 
296  // SoundArchive の初期化
297  const char* archiveAbsolutePath = nns::atk::GetAbsolutePath(ArchiveRelativePath);
298  isSuccess = g_SoundArchive.Open(archiveAbsolutePath);
299  NN_ABORT_UNLESS( isSuccess, "cannot open SoundArchive(%s)\n", archiveAbsolutePath );
300 
301  // SoundArchive のパラメータ情報をメモリにロード
302  std::size_t infoBlockSize = g_SoundArchive.GetHeaderSize();
303  g_pMemoryForInfoBlock = nns::atk::Allocate( infoBlockSize, nn::atk::FsSoundArchive::BufferAlignSize );
304  isSuccess = g_SoundArchive.LoadHeader( g_pMemoryForInfoBlock, infoBlockSize );
305  NN_ABORT_UNLESS( isSuccess, "cannot load InfoBlock" );
306 
307  // SoundDataManager の初期化
308  std::size_t memSizeForSoundDataManager = g_SoundDataManager.GetRequiredMemSize( &g_SoundArchive );
309  g_pMemoryForSoundDataManager = nns::atk::Allocate( memSizeForSoundDataManager, nn::atk::SoundDataManager::BufferAlignSize );
310  isSuccess = g_SoundDataManager.Initialize(
311  &g_SoundArchive,
312  g_pMemoryForSoundDataManager,
313  memSizeForSoundDataManager );
314  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundDataManager" );
315 
316  // SoundArchivePlayer で用いるストリームバッファの初期化
317  // ストリームバッファはメモリプール管理されているヒープから確保する必要があります。
318  std::size_t memSizeForStreamBuffer = g_SoundArchivePlayer.GetRequiredStreamBufferSize( &g_SoundArchive );
319  g_pMemoryForStreamBuffer = nns::atk::AllocateForMemoryPool(memSizeForStreamBuffer);
320 
321  // 専用のヒープをメモリプールにアタッチ
322  nn::atk::SoundSystem::AttachMemoryPool(&g_MemoryPool, nns::atk::GetPoolHeapAddress(), nns::atk::GetPoolHeapSize());
323 
324  // SoundArchivePlayer の初期化
325  std::size_t memSizeForSoundArchivePlayer = g_SoundArchivePlayer.GetRequiredMemSize( &g_SoundArchive );
326  g_pMemoryForSoundArchivePlayer = nns::atk::Allocate( memSizeForSoundArchivePlayer, nn::atk::SoundArchivePlayer::BufferAlignSize);
327  isSuccess = g_SoundArchivePlayer.Initialize(
328  &g_SoundArchive,
329  &g_SoundDataManager,
330  g_pMemoryForSoundArchivePlayer, memSizeForSoundArchivePlayer,
331  g_pMemoryForStreamBuffer, memSizeForStreamBuffer );
332  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundArchivePlayer" );
333 
334  // EffectReverb をカスタムサブミックスに追加
335  {
336  g_EffectReverb.SetSampleRate(nn::atk::EffectBase::SampleRate_48000);
337  g_EffectReverb.SetChannelMode(nn::atk::EffectBase::ChannelMode_2Ch);
338  g_EffectReverb.SetEarlyMode(nn::atk::EffectReverb::EarlyMode_Hall);
339  g_EffectReverb.SetLateMode(nn::atk::EffectReverb::LateMode_Hall);
340  g_EffectReverb.SetLateGain(0.6f);
341  g_EffectReverb.SetDecayTime(nn::TimeSpan::FromMilliSeconds(3000));
342 
343  // SubMix[A] に接続してバスのセンドの確認に用います。
344  std::size_t memSizeForEffectReverb = g_EffectReverb.GetRequiredMemSize();
345  g_pMemoryForEffectReverb = nns::atk::AllocateForMemoryPool( memSizeForEffectReverb, nn::audio::BufferAlignSize );
346  g_EffectReverb.SetEnabled( true );
347 
348  // SubMix[A] の AppendEffectBusIndex バスにエフェクトを追加
349  g_SubMix[SubMixIndex_A].AppendEffect( &g_EffectReverb, AppendEffectBusIndex, g_pMemoryForEffectReverb, memSizeForEffectReverb );
350  }
351 }

353 void FinalizeAtk()
354 {
355  // Effect をカスタムサブミックスから削除
356  {
357  const auto WaitTime = nn::TimeSpan::FromMilliSeconds( 5 );
358  g_EffectReverb.SetEnabled( false );
359  while( !g_EffectReverb.IsRemovable() )
360  {
361  nn::os::SleepThread( WaitTime );
362  }
363  g_SubMix[SubMixIndex_A].ClearEffect( AppendEffectBusIndex );
364  }
365 
366  g_SoundArchivePlayer.Finalize();
367 
368  // 専用のヒープをメモリプールからデタッチ
369  nn::atk::SoundSystem::DetachMemoryPool(&g_MemoryPool);
370 
371  g_SoundDataManager.Finalize();
372  g_SoundArchive.Close();
373  g_SoundHeap.Destroy();
374 
375  // SubMix の終了処理
376  FinalizeSubMix();
377 
378  nn::atk::SoundSystem::Finalize();
379 
380  nns::atk::FreeForMemoryPool(g_pMemoryForStreamBuffer);
381  nns::atk::FreeForMemoryPool(g_pMemoryForEffectReverb);
382  nns::atk::Free(g_pMemoryForSoundArchivePlayer);
383  nns::atk::Free(g_pMemoryForSoundDataManager);
384  nns::atk::Free(g_pMemoryForInfoBlock);
385  nns::atk::Free(g_pMemoryForSoundHeap);
386  nns::atk::Free(g_pMemoryForSubMix[SubMixIndex_C]);
387  nns::atk::Free(g_pMemoryForSubMix[SubMixIndex_B]);
388  nns::atk::Free(g_pMemoryForSubMix[SubMixIndex_A]);
389  nns::atk::Free(g_pMemoryForSoundSystem);
390 }

403 void PlayWithStartSound(nn::atk::SoundArchive::ItemId soundId, const char* debugLabelName)
404 {
405  g_SoundHandle.Stop( 0 );
406 
407  nn::atk::SoundArchivePlayer::StartInfo info;
408  info.enableFlag |= nn::atk::SoundArchivePlayer::StartInfo::EnableFlagBit_OutputReceiver;
409  info.pOutputReceiver = &g_SubMix[g_TargetSubMixIndex]; // 選択中のサブミックスで再生します
410 
411  bool result = g_SoundArchivePlayer.StartSound( &g_SoundHandle, soundId, &info ).IsSuccess();
412  NNS_LOG("StartSound(%s) on SubMix[%d] ... (%d)\n", debugLabelName, g_TargetSubMixIndex, result);
413 
414  // SubMix[A] のエフェクトが確認できるようにセンドを設定します
415  g_SoundHandle.SetSend(AppendEffectBusIndex, 0.5f);
416  // SoundMaker によりセンドMainにデフォルト値が設定されているため、そのセンド量を減らします
417  g_SoundHandle.SetSend(0, -0.5f);
418 }

442 bool UpdateAtk()
443 {
444  // StartSound / Stop
445  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::A >() )
446  {
447  PlayWithStartSound(SEQ_MARIOKART, "SEQ_MARIOKART");
448  }
449  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::X >() )
450  {
451  PlayWithStartSound(SE_YOSHI, "SE_YOSHI");
452  }
453  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Y >() )
454  {
455  PlayWithStartSound(STRM_MARIOKART, "STRM_MARIOKART");
456  }
457  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::B >() )
458  {
459  g_SoundHandle.Stop( 0 );
460  }
461 
462  bool isTargetSubMixChanged = false;
463  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Left >() || nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Right >() )
464  {
465  g_TargetSubMixIndex += nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Left >() ? -1 : 1;
466  if( g_TargetSubMixIndex < 0 )
467  {
468  g_TargetSubMixIndex = SubMixIndex_Count - 1;
469  }
470  if( g_TargetSubMixIndex >= SubMixIndex_Count )
471  {
472  g_TargetSubMixIndex = 0;
473  }
474  isTargetSubMixChanged = true;
475  }
476 
477  // Print Usage
478  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::L >() )
479  {
480  PrintUsage();
481  }
482 
483  if( isTargetSubMixChanged )
484  {
485  PrintTargetSubMixState();
486  }
487 
488  // Exit
489  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Start >() )
490  {
491  return false;
492  }
493 
494  g_SoundArchivePlayer.Update();
495 
496  return true;
497 } //NOLINT(impl/function_size)

サンプルプログラムの解説

先のサンプルプログラムの Atk に関連している処理の流れは以下の通りです(AtkSimple と違う部分には追加要素と書かれています)。

  • アロケータの初期化
    • 内部で２種類（メモリプール管理対象用とそれ以外）のアロケータを初期化。
  • Atk ライブラリの初期化 (InitializeAtk())
    • SoundSystem の初期化パラメータの設定 (追加要素)
    • SoundSystem の初期化
    • SubMix の初期化 (InitializeSubMix()) (追加要素)
    • SoundHeap の初期化
    • SoundArchive をオープンし、パラメータ情報をロード
    • SoundDataManager の初期化
    • SoundArchivePlayer で用いるストリームバッファをメモリプール管理対象用アロケータから確保し初期化
    • メモリプール管理対象のメモリをメモリプールにアタッチ
    • SoundArchivePlayer の初期化
    • SubMix にエフェクトを追加 (追加要素)
  • SoundHeap にウェーブサウンドとシーケンスサウンドの再生に必要な情報をロード (LoadData())
  • メインループでの処理
    • キー操作による処理
      • サウンドの再生・停止処理を実行
        • サウンドの再生時に使用するサブミックスを設定 (追加要素)
      • 使用するサブミックスの選択 (追加要素)
    • SoundArchivePlayer の更新
  • Atk ライブラリの終了処理 (FinalizeAtk())
    • SubMix からエフェクトを削除 (追加要素)
    • SoundArchivePlayer の 終了処理
    • メモリプール管理対象のメモリをメモリプールからデタッチ
    • その他の終了処理
    • SubMix の終了処理 (追加要素)
    • SoundSystemの終了処理
    • メモリ解放
  • アロケータの終了処理

SubMix の初期化・終了処理は、それぞれ SoundSystem の初期化直後・終了処理直前に実行する必要があります。 それ以外のタイミングでは安全に処理できない点にご注意ください。
