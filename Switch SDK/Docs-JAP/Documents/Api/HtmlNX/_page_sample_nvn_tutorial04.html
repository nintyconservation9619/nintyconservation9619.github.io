Overview

This tutorial builds off of Tutorial03 and introduces asset file loading in parallel on multiple threads and rendering multiple models with different asset data and shaders. Three cubes are rendered, one with a simple texture, one with a mipmapped texture, and one with a cubemap. The multi threaded loading is shown in the MultiThreadedLoadfunction of this tutorial class and in MultiThreadAssetFileLoadingHelper.cpp/.h.

603 void MultiThreadedAssetFileLoading::MultiThreadedLoad()
604 {
605  AssetLoaderArg* pArgs[3];
606  nn::os::MutexType poolMutex;
607  nn::os::InitializeMutex(&poolMutex, false, 0);
608 
609  pArgs[0] = new AssetLoaderArg(&m_Device, "multiThreadedAssetFile0.out", &poolMutex, m_pTextureIDManager, false, false);
610  pArgs[1] = new AssetLoaderArg(&m_Device, "multiThreadedAssetFile1.out", &poolMutex, m_pTextureIDManager, true, false);
611  pArgs[2] = new AssetLoaderArg(&m_Device, "multiThreadedAssetFile2.out", &poolMutex, m_pTextureIDManager, false, true);
612 
613  MultiThreadedAssetFileLoadingHelper* pLoaders[3];
614  pLoaders[0] = new MultiThreadedAssetFileLoadingHelper(pArgs[0]);
615  pLoaders[1] = new MultiThreadedAssetFileLoadingHelper(pArgs[1]);
616  pLoaders[2] = new MultiThreadedAssetFileLoadingHelper(pArgs[2]);
617 
618  nn::os::ThreadType hThreadArray[3];
619 
620  size_t cacheSize = 0;
621  nn::Result result = nn::fs::QueryMountRomCacheSize(&cacheSize);
622  NN_ASSERT( result.IsSuccess() );
623 
624  char* mountRomCacheBuffer = new(std::nothrow) char[cacheSize];
625  NN_ASSERT_NOT_NULL(mountRomCacheBuffer);

951 /*

965  nvnQueueFenceSync(&m_Queue, &m_CommandBufferSync, NVN_SYNC_CONDITION_ALL_GPU_COMMANDS_COMPLETE, 0);

982 /*
983  * MultiThreadedAssetFileLoading::Resize
984  * -------------------------------------
985  * This method is called everytime the window is resized and is passed
986  * the new size of the window. It frees the old render target and creates a new
987  * render target with the new screen size. A new set of commands for the command buffer
988  * is recorded.
989  */
990 void MultiThreadedAssetFileLoading::Resize(int width, int height)
991 {
992  /* Check for the window being minimized or having no visible surface. */
993  if(width == 0 || height == 0)
994  {
995  return;
996  }
997 
998  /* Save the current height and width of the screen. */
999  m_ScreenHeight = height;
1000  m_ScreenWidth = width;
1001 
1002  /* Sync to safely rewrite the command buffer. */
1003  nvnQueueFenceSync(&m_Queue, &m_CommandBufferSync, NVN_SYNC_CONDITION_ALL_GPU_COMMANDS_COMPLETE, 0);
1004  nvnQueueFlush(&m_Queue);
1005  nvnSyncWait(&m_CommandBufferSync, NVN_WAIT_TIMEOUT_MAXIMUM);
1006 
1007  /* If it's the first time Resize is called, allocate the NVNwindow. */
1008  if (m_pWindow == NULL)
1009  {
1010  m_pWindow = new NVNwindow;
1011  }
1012  /*
1013  * Otherwise finalize (free) the NVNwindow used for the previous window size.
1014  * The NVNWindow must be finalized before a render target it owns is finalized.
1015  */
1016  else
1017  {
1018  nvnWindowFinalize(m_pWindow);
1019  }
1020 
1021  /* Set up the builder for the render target. */
1022  nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
1023  nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_DISPLAY_BIT | NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
1024  nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
1025  nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
1026  nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_RGBA8);
1027 
1028  for(int i = 0; i < g_NumColorBuffers; ++i)
1029  {
1030  /* If it's the first time Resize is called, allocate the render target. */
1031  if (m_pRenderTargets[i] == NULL)
1032  {
1033  m_pRenderTargets[i] = new NVNtexture;
1034  }
1035  /* Otherwise finalize (free) the render target used for the previous window size. */
1036  else
1037  {
1038  nvnTextureFinalize(m_pRenderTargets[i]);
1039  }
1040 
1041  nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * i);
1042 
1043  /* Create the texture using the current state of the texture builder. */
1044  if (!nvnTextureInitialize(m_pRenderTargets[i], &m_RenderTargetBuilder))
1045  {
1046  NN_ASSERT(0, "Failed to initialize render target");
1047  }
1048  }
1049 
1050  /* If it's the first time Resize is called, allocate the depth buffer. */
1051  if (m_pDepthBuffer == NULL)
1052  {
1053  m_pDepthBuffer = new NVNtexture;
1054  }
1055  /* Otherwise finalize (free) the depth buffer used for the previous window size. */
1056  else
1057  {
1058  nvnTextureFinalize(m_pDepthBuffer);
1059  }
1060 
1061  /* Initialize depth buffer for render target. */
1062  nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
1063  nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
1064  nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
1065  nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
1066  nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_DEPTH32F);
1067  nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * g_NumColorBuffers);
1068 
1069  if (!nvnTextureInitialize(m_pDepthBuffer, &m_RenderTargetBuilder))
1070  {
1071  NN_ASSERT(0, "Failed to initialize depth target");
1072  }
1073 
1074  /* Pass off the render targets to the window. */
1075  nvnWindowBuilderSetTextures(&m_WindowBuilder, g_NumColorBuffers, m_pRenderTargets);
1076  nvnWindowInitialize(m_pWindow, &m_WindowBuilder);
1077 
1078  /* Reset the command buffer memory. */
1079  nvnCommandBufferAddCommandMemory(&m_CommandBuffer, m_pCommandMemoryPool->GetMemoryPool(), m_CommandPoolOffset, g_CommandMemorySize);
1080  nvnCommandBufferAddControlMemory(&m_CommandBuffer, m_pControlPool, g_ControlMemorySize);
1081 
1082 
1083  /* Starts the recording of a new set of commands for the given command buffer. */
1084  nvnCommandBufferBeginRecording(&m_CommandBuffer);
1085  {
1086  /* Bind the texture and sampler descriptor pools. */
1087  m_pTextureIDManager->SetSamplerPool(&m_CommandBuffer);
1088  m_pTextureIDManager->SetTexturePool(&m_CommandBuffer);
1089 
1090  /* Sets the scissor rectangle and viewport to the full screen */
1091  nvnCommandBufferSetScissor(&m_CommandBuffer, 0, 0, m_ScreenWidth, m_ScreenHeight);
1092  nvnCommandBufferSetViewport(&m_CommandBuffer, 0, 0, m_ScreenWidth, m_ScreenHeight);
1093 
1094  /* Clears the currently set render target at a given index. */
1095  float clearColor[4] = { 0.4f, 0.55f, 0.6f, 1.0f };
1096  nvnCommandBufferClearColor(&m_CommandBuffer, 0, clearColor, NVN_CLEAR_COLOR_MASK_RGBA);
1097  nvnCommandBufferClearDepthStencil(&m_CommandBuffer, 1.0, NVN_TRUE, 0, 0);
1098 
1099  /* Bind the render state objects. */
1100  nvnCommandBufferBindBlendState(&m_CommandBuffer, &m_BlendState);
1101  nvnCommandBufferBindChannelMaskState(&m_CommandBuffer, &m_ChannelMaskState);
1102  nvnCommandBufferBindColorState(&m_CommandBuffer, &m_ColorState);
1103  nvnCommandBufferBindDepthStencilState(&m_CommandBuffer, &m_DepthStencilState);
1104  nvnCommandBufferBindMultisampleState(&m_CommandBuffer, &m_MultiSampleState);
1105  nvnCommandBufferBindPolygonState(&m_CommandBuffer, &m_PolygonState);
1106  nvnCommandBufferSetSampleMask(&m_CommandBuffer, static_cast<uint32_t>(~0));
1107 
1108  NVNbufferAddress vertBufferAddresses[3] = {m_BasicCubeUniformVertexBuffer.GetBufferAddress(),
1109  m_MipmapCubeUniformVertexBuffer.GetBufferAddress(),
1110  m_CubeMapUniformVertexBuffer.GetBufferAddress()};
1111  NVNbufferAddress fragBufferAddresses[3] = {m_BasicCubeUniformFragmentBuffer.GetBufferAddress(),
1112  m_MipmapCubeUniformFragmentBuffer.GetBufferAddress(),
1113  m_CubeMapUniformFragmentBuffer.GetBufferAddress()};
1114 
1115  /* Render each object in a loop. */
1116  for(int i = 0; i < g_CubesToRender; ++i)
1117  {
1118  std::vector<NVNModelData*>& modelData = m_DataHolders[i]->GetModelData();
1119  Model* pModel = &modelData[0]->m_Model;
1120 
1121  /* Bind the vertex buffer(s). */
1122  NVNbufferAddress vboAddr = nvnBufferGetAddress(&modelData[0]->m_VertexBuffer);
1123  for (size_t j = 0; j < pModel->m_VertexAttributes.size(); ++j)
1124  {
1125  VertexAttribute& attr = pModel->m_VertexAttributes[j];
1126  nvnCommandBufferBindVertexBuffer(&m_CommandBuffer, attr.m_Location, vboAddr + modelData[0]->m_VertexAttributeBufferOffsets[j], attr.m_DataSize);
1127  }
1128 
1129  /* Bind the vertex states. */
1130  nvnCommandBufferBindVertexAttribState(&m_CommandBuffer, static_cast<int>(modelData[0]->m_VertexAttributeStates.size()), &modelData[0]->m_VertexAttributeStates[0]);
1131  nvnCommandBufferBindVertexStreamState(&m_CommandBuffer, static_cast<int>(modelData[0]->m_VertexStreamStates.size()), &modelData[0]->m_VertexStreamStates[0]);
1132 
1133  /* Bind the shader program and uniforms. */
1134  std::vector<NVNProgramData*>& programData = m_DataHolders[i]->GetProgramData();
1135  ShaderTypes::ShaderType shaderType = programData[0]->m_ShaderType;
1136  uint32_t vertexUniformBlockBindingLocation = ShaderTypes::BlockVS_GetBinding(NVN_SHADER_STAGE_VERTEX, shaderType);
1137  uint32_t vertexUniformBlockSize = ShaderTypes::BlockVS_GetUniformBlockSize(shaderType);
1138  nvnCommandBufferBindUniformBuffer(&m_CommandBuffer,
1139  NVN_SHADER_STAGE_VERTEX,
1140  vertexUniformBlockBindingLocation,
1141  vertBufferAddresses[i],
1142  vertexUniformBlockSize);

Sample Program Description

This tutorial renders three cubes: one that is the same as Tutorial03, and second cube that demostrates mipmapping, and a third that is textured with a cube map. The mipmapped cube has different color for each mip level so that it's easy to see at what points the levels switch. The mipmapped texture was built using a texture file per mip level by Tutorial02. The mipmapped cube and the basic cube are rendered with the same shader. The third cube is textured with a cube map built out of six textures and rendered using a second shader program.
