31 void ReferUtf16ToSjisTable(unsigned char *pDst, uint16_t src)
32 {
33  unsigned char utf16Upper = static_cast<unsigned char>(src >> 8); // 探索する変換元の UTF-16 上位 8 bit
34  unsigned char tableIndex = static_cast<unsigned char>(src & 0xFFu);
35 
36  // g_pUtf16ToSjisTable の UTF-16 の下位 8bit が一致する範囲を二分探索
37  int begin = g_pUtf16ToSjisIndex[tableIndex];
38  int end = g_pUtf16ToSjisIndex[tableIndex + 1] - 1;
39  const unsigned char* pMiddle;
40 
41  while (begin <= end)
42  {
43  int middleIndex = (begin + end) / 2;
44  int middleVal = g_pUtf16ToSjisTable[middleIndex * 3];
45  pMiddle = &g_pUtf16ToSjisTable[middleIndex * 3];
46 
47  if (*pMiddle == utf16Upper)
48  {
49  pDst[0] = *(++pMiddle);
50  pDst[1] = *(++pMiddle);
51  return;
52  }
53 
54  if (middleVal > utf16Upper)
55  {
56  end = middleIndex - 1;
57  }
58  else if (middleVal < utf16Upper)
59  {
60  begin = middleIndex + 1;
61  }
62  }
63 
64  // 変換できなかった
65  pDst[0] = pDst[1] = 0;
66 }

68 //
69 // UTF-16 1 文字を Shift_JIS に変換します。
70 //
71 void ConvertCharacterUnicodeToSjis(unsigned char *pDst, uint16_t src)
72 {
73  if (src < 0x0080u)
74  {
75  // 制御文字と ASCII
76  pDst[0] = static_cast<unsigned char>(src);
77  pDst[1] = 0x00u;
78  return;
79  }
80  if (src >= 0xFF61u && src <= 0xFF9Fu)
81  {
82  // 半角カナ
83  pDst[0] = static_cast<unsigned char>(src - 0xFEC0u);
84  pDst[1] = 0;
85  return;
86  }
87  if (src < 0xD800u || src >= 0xF900u)
88  {
89  // ひらがな、カタカナ、 CJK 統合漢字は変換テーブルから引く
90  ReferUtf16ToSjisTable(pDst, src);
91  return;
92  }
93  if (src >= 0xE000u && src <= 0xE757u)
94  {
95  // Private Use Area
96  // Shift_JIS の外字コードに変換
97  uint32_t modNum = (src - 0xE000u) % (94 * 2);
98  uint32_t divNum = (src - 0xE000u - modNum) / (94 * 2);
99  pDst[0] = static_cast<unsigned char>(divNum + 0xF0u);
100  pDst[1] = static_cast<unsigned char>(modNum < 0x3Fu ? modNum + 0x40u : modNum + 0x80u - 0x3Fu);
101  return;
102  }
103  if (src >= 0xF8F0 && src <= 0xF8F3)
104  {
105  // Private Use Area の一部を Windows の WideCharToMultiByte の挙動に合わせる
106  if (src == 0xF8F0)
107  {
108  pDst[0] = 0xA0;
109  }
110  else
111  {
112  pDst[0] = static_cast<unsigned char>(src - 0xF8F1 + 0xFD);
113  }
114  pDst[1] = 0;
115  return;
116  }
117  // 変換できなかった
118  pDst[0] = pDst[1] = 0x00u;
119 }

126 nn::util::CharacterEncodingResult ConvertStringSjisToUtf16(uint16_t *pDst, int dstLength, const char *pSrc)
127 {
128  NN_ASSERT_NOT_NULL(pDst);
129  NN_ASSERT_NOT_NULL(pSrc);
130  NN_ASSERT_GREATER_EQUAL(dstLength, 0);
131 
132  int index = 0;
133 
134  while (*pSrc != '\0')
135  {
136  unsigned char sjisUpper = static_cast<unsigned char>(*pSrc);
137  unsigned char sjisLower;
138  uint16_t utf16Char;
139 
140  if (index >= dstLength)
141  {
142  return nn::util::CharacterEncodingResult_InsufficientLength;
143  }
144 
145  if ((sjisUpper >= 0x81u && sjisUpper <= 0x9Fu) || (sjisUpper >= 0xE0u && sjisUpper <= 0xFCu))
146  {
147  // JIS 漢字の範囲はテーブルを利用して変換
148  ++pSrc;
149  sjisLower = static_cast<unsigned char>(*pSrc);
150  if (sjisLower == '\0')
151  {
152  return nn::util::CharacterEncodingResult_InvalidFormat;
153  }
154  if ((sjisLower >= 0x40u && sjisLower <= 0x7Eu) || (sjisLower >= 0x80u && sjisLower <= 0xFCu))
155  {
156  if (sjisUpper < 0xF0u)
157  {
158  // JIS 漢字
159  int tableIndex = 0;
160  tableIndex += 94 * 2 * (sjisUpper < 0xA0u ? sjisUpper - 0x81u : sjisUpper - 0xE0u + 0x1Fu);
161  tableIndex += (sjisLower < 0x7Fu ? sjisLower - 0x40u : sjisLower - 0x80u + 0x3Fu);
162  utf16Char = g_pSjisToUtf16Table[tableIndex];
163  }
164  else if (sjisUpper >= 0xFAu)
165  {
166  // IBM 拡張文字
167  int tableIndex = 0;
168  tableIndex += 94 * 2 * (sjisUpper - 0xE0u + 0x1Fu - 0x0Au);
169  tableIndex += (sjisLower < 0x7Fu ? sjisLower - 0x40u : sjisLower - 0x80u + 0x3Fu);
170  utf16Char = g_pSjisToUtf16Table[tableIndex];
171  }
172  else
173  {
174  // 外字コード
175  // Unicode の Private Use Area に変換
176  int utf16Index = 0;
177  utf16Index += 94 * 2 * (sjisUpper - 0xF0u);
178  utf16Index += (sjisLower < 0x7Fu ? sjisLower - 0x40u : sjisLower - 0x80u + 0x3Fu);
179  utf16Char = static_cast<uint16_t>(0xE000u + utf16Index);
180  }
181  }
182  else
183  {
184  return nn::util::CharacterEncodingResult_InvalidFormat;
185  }
186  }
187  else
188  {
189  if (sjisUpper <= 0x80u)
190  {
191  // Windows の MultiByteToWideChar に合わせ 0x80 も変換可能とする
192  utf16Char = static_cast<uint16_t>(sjisUpper);
193  }
194  else if (sjisUpper == 0xA0u)
195  {
196  // Windows の MultiByteToWideChar に合わせ 0xA0 も変換可能とする
197  utf16Char = 0xF8F0u;
198  }
199  else if (sjisUpper < 0xE0u)
200  {
201  // 半角カナ
202  utf16Char = static_cast<uint16_t>(sjisUpper + 0xFEC0u);
203  }
204  else if (sjisUpper >= 0xFDu)
205  {
206  // Windows の MultiByteToWideChar に合わせ 0xFDXX-0xFFXX も変換可能とする
207  utf16Char = static_cast<uint16_t>(0xF8F1u + (sjisUpper - 0xFDu));
208  }
209  else
210  {
211  return nn::util::CharacterEncodingResult_InvalidFormat;
212  }
213  }
214 
215  if (utf16Char == 0x0000u)
216  {
217  return nn::util::CharacterEncodingResult_InvalidFormat;
218  }
219  *pDst = utf16Char;
220  ++pDst;
221  ++index;
222  ++pSrc;
223  }
224  return nn::util::CharacterEncodingResult_Success;
225 }

サンプルプログラムの解説

Shift_JIS 文字列と UTF-16 文字列や UTF-8 文字列を相互変換します。 本サンプルでは厳密な Shift_JIS ではなく、 Shift_JIS を拡張した Windows コードページ 932 で扱われている符号化文字集合を変換可能としています。 特に Shift_JIS と UTF-16 の変換においては、変換が成功する場合 Windows の WideCharToMultiByte() 関数と MultiByteToWideChar() 関数と同等の変換結果が得られるようになっています。 変換失敗時の挙動についてはこの限りではありません。

Shift_JIS と UTF-16 の変換においては変換テーブルを持ちそれを利用し変換を行っています。 Shift_JIS と UTF-8 の変換においては間に UTF-16 と UTF-8 の変換を挟むことで実現しています。 UTF-16 と UTF-8 の変換には nn::util::ConvertStringUtf16NativeToUtf8() 関数と nn::util::ConvertStringUtf8ToUtf16Native() 関数を利用しています。
