nn::atk ライブラリのエフェクトのサンプルプログラムの解説です。

概要

Atk ライブラリのエフェクトを適用して、音声を再生するサンプルです。 Delay や Reverb のエフェクト、またユーザ定義のエフェクトを適用することができます。

ファイル構成

本サンプルプログラムは Samples/Sources/Applications/AtkEffect 以下にあります。

必要な環境

とくになし

操作方法

コンソールに操作方法が表示されます。

注意事項

実行中のログが、画面上とコンソールに表示され、 起動直後のログとして操作方法が表示されます。

PC 版ではウィンドウは立ち上がらず、コンソール表示のみになる点にご注意ください。

実行手順

サンプルプログラムをビルドし、実行してください。

175 void AppendEffect( int effectType )
176 {
177  switch ( effectType )
178  {
179  case EffectType_Delay:
180  {
181  g_Delay.SetSampleRate(nn::atk::EffectBase::SampleRate_48000);
182  g_Delay.SetChannelMode(nn::atk::EffectBase::ChannelMode_2Ch);
183  g_Delay.SetDelayTimeMax(nn::TimeSpan::FromMilliSeconds(3000));
184  g_Delay.SetFeedbackGain(0.6f);
185  g_Delay.SetEnabled(true);
186 
187  size_t delayBufferSize = g_Delay.GetRequiredMemSize();
188  // エフェクト用のバッファはメモリプール管理されているヒープから確保する必要があります。
189  g_pMemoryForEffectDelay = nns::atk::AllocateForMemoryPool(delayBufferSize, nn::audio::BufferAlignSize);
190  NN_ABORT_UNLESS_NOT_NULL(g_pMemoryForEffectDelay);
191 
192  bool isSuccess = nn::atk::SoundSystem::AppendEffect(nn::atk::AuxBus_A, &g_Delay, g_pMemoryForEffectDelay, delayBufferSize);
193  NNS_LOG( "Delay %d\n", isSuccess );
194  }
195  break;
196  case EffectType_MonoRightDelay:
197  {
198  g_MonoDelay.SetSampleRate(nn::atk::EffectBase::SampleRate_48000);
199  const nn::atk::ChannelIndex MonoDelayChannel[] = {nn::atk::ChannelIndex_FrontRight};
200  g_MonoDelay.SetChannelIndex(MonoDelayChannel, nn::atk::EffectBase::ChannelMode_1Ch);
201  g_MonoDelay.SetDelayTimeMax(nn::TimeSpan::FromMilliSeconds(3000));
202  g_MonoDelay.SetDelayTime(nn::TimeSpan::FromMilliSeconds(400));
203  g_MonoDelay.SetFeedbackGain(0.6f);
204  g_MonoDelay.SetEnabled(true);
205 
206  size_t monoDelayBufferSize = g_MonoDelay.GetRequiredMemSize();
207  // エフェクト用のバッファはメモリプール管理されているヒープから確保する必要があります。
208  g_pMemoryForEffectMonoDelay = nns::atk::AllocateForMemoryPool(monoDelayBufferSize, nn::audio::BufferAlignSize);
209  NN_ABORT_UNLESS_NOT_NULL(g_pMemoryForEffectMonoDelay);
210 
211  bool isSuccess = nn::atk::SoundSystem::AppendEffect(nn::atk::AuxBus_A, &g_MonoDelay, g_pMemoryForEffectMonoDelay, monoDelayBufferSize);
212  NNS_LOG( "Delay(Mono Right) %d\n", isSuccess );
213  }
214  break;
215  case EffectType_Reverb:
216  {
217  g_Reverb.SetSampleRate(nn::atk::EffectBase::SampleRate_48000);
218  g_Reverb.SetChannelMode(nn::atk::EffectBase::ChannelMode_2Ch);
219  g_Reverb.SetEarlyMode(nn::atk::EffectReverb::EarlyMode_Hall);
220  g_Reverb.SetLateMode(nn::atk::EffectReverb::LateMode_Hall);
221  g_Reverb.SetLateGain(0.6f);
222  g_Reverb.SetDecayTime(nn::TimeSpan::FromMilliSeconds(3000));
223  g_Reverb.SetEnabled(true);
224 
225  size_t reverbBufferSize = g_Reverb.GetRequiredMemSize();
226  // エフェクト用のバッファはメモリプール管理されているヒープから確保する必要があります。
227  g_pMemoryForEffectReverb = nns::atk::AllocateForMemoryPool(reverbBufferSize, nn::audio::BufferAlignSize);
228  NN_ABORT_UNLESS_NOT_NULL(g_pMemoryForEffectReverb);
229 
230  bool isSuccess = nn::atk::SoundSystem::AppendEffect(nn::atk::AuxBus_A, &g_Reverb, g_pMemoryForEffectReverb, reverbBufferSize);
231  NNS_LOG( "Reverb %d\n", isSuccess );
232  }
233  break;
234  case EffectType_I3dl2Reverb:
235  {
236  g_I3dl2Reverb.SetSampleRate(nn::atk::EffectBase::SampleRate_48000);
237  g_I3dl2Reverb.SetChannelMode(nn::atk::EffectBase::ChannelMode_6Ch);
238  g_I3dl2Reverb.SetEnabled(true);
239 
240  size_t i3dl2ReverbBufferSize = g_I3dl2Reverb.GetRequiredMemSize();
241  // エフェクト用のバッファはメモリプール管理されているヒープから確保する必要があります。
242  g_pMemoryForEffectI3dl2Reverb = nns::atk::AllocateForMemoryPool(i3dl2ReverbBufferSize, nn::audio::BufferAlignSize);
243  NN_ABORT_UNLESS_NOT_NULL(g_pMemoryForEffectI3dl2Reverb);
244  bool isSuccess = nn::atk::SoundSystem::AppendEffect(nn::atk::AuxBus_A, &g_I3dl2Reverb, g_pMemoryForEffectI3dl2Reverb, i3dl2ReverbBufferSize);
245  NNS_LOG( "I3DL2 Reverb %d\n", isSuccess );
246  }
247  break;
248  case EffectType_Aux:
249  {
250  g_EffectVolumeDown.SetAudioFrameCount(5);
251  g_EffectVolumeDown.SetChannelCount(2);
252  g_EffectVolumeDown.SetEnabled(true);
253  g_pAux = &g_EffectVolumeDown;
254 
255  size_t auxBufferSize = nn::atk::SoundSystem::GetRequiredEffectAuxBufferSize(g_pAux);
256  // エフェクト用のバッファはメモリプール管理されているヒープから確保する必要があります。
257  g_pMemoryForEffectAux = nns::atk::AllocateForMemoryPool(auxBufferSize, nn::audio::BufferAlignSize);
258  NN_ABORT_UNLESS_NOT_NULL(g_pMemoryForEffectAux);
259 
260  bool isSuccess = nn::atk::SoundSystem::AppendEffect(nn::atk::AuxBus_A, g_pAux, g_pMemoryForEffectAux, auxBufferSize);
261  NNS_LOG("Aux %d (bufferSize = %zu)\n", isSuccess, auxBufferSize);
262  }
263  break;
264  case EffectType_None:
265  NNS_LOG("None\n");
266  break;
267  default:
268  NN_UNEXPECTED_DEFAULT;
269  break;
270  }

369 void InitializeAtk()
370 {
371  bool isSuccess = true;
372 
373  // SoundSystem の初期化
374  nn::atk::SoundSystem::SoundSystemParam param;
375  std::size_t memSizeForSoundSystem = nn::atk::SoundSystem::GetRequiredMemSize( param );
376  g_pMemoryForSoundSystem = nns::atk::Allocate( memSizeForSoundSystem, nn::atk::SoundSystem::WorkMemoryAlignSize );
377  nn::atk::SoundSystem::Initialize(
378  param,
379  reinterpret_cast<uintptr_t>( g_pMemoryForSoundSystem ),
380  memSizeForSoundSystem );
381 
382  // SoundHeap の初期化
383  g_pMemoryForSoundHeap = nns::atk::Allocate( SoundHeapSize );
384  isSuccess = g_SoundHeap.Create( g_pMemoryForSoundHeap, SoundHeapSize );
385  NN_ABORT_UNLESS( isSuccess, "cannot create SoundHeap" );
386 
387  // SoundArchive の初期化
388  const char* archiveAbsolutePath = nns::atk::GetAbsolutePath(ArchiveRelativePath);
389  isSuccess = g_SoundArchive.Open(archiveAbsolutePath);
390  NN_ABORT_UNLESS( isSuccess, "cannot open SoundArchive(%s)\n", archiveAbsolutePath );
391 
392  // SoundArchive のパラメータ情報をメモリにロード
393  std::size_t infoBlockSize = g_SoundArchive.GetHeaderSize();
394  g_pMemoryForInfoBlock = nns::atk::Allocate( infoBlockSize, nn::atk::FsSoundArchive::BufferAlignSize );
395  isSuccess = g_SoundArchive.LoadHeader( g_pMemoryForInfoBlock, infoBlockSize );
396  NN_ABORT_UNLESS( isSuccess, "cannot load InfoBlock" );
397 
398  // SoundDataManager の初期化
399  std::size_t memSizeForSoundDataManager = g_SoundDataManager.GetRequiredMemSize( &g_SoundArchive );
400  g_pMemoryForSoundDataManager = nns::atk::Allocate( memSizeForSoundDataManager, nn::atk::SoundDataManager::BufferAlignSize );
401  isSuccess = g_SoundDataManager.Initialize(
402  &g_SoundArchive,
403  g_pMemoryForSoundDataManager,
404  memSizeForSoundDataManager );
405  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundDataManager" );
406 
407  // SoundArchivePlayer で用いるストリームバッファの初期化
408  // ストリームバッファはメモリプール管理されているヒープから確保する必要があります。
409  std::size_t memSizeForStreamBuffer = g_SoundArchivePlayer.GetRequiredStreamBufferSize( &g_SoundArchive );
410  g_pMemoryForStreamBuffer = nns::atk::AllocateForMemoryPool(memSizeForStreamBuffer);
411 
412  // 専用のヒープをメモリプールにアタッチ
413  nn::atk::SoundSystem::AttachMemoryPool(&g_MemoryPool, nns::atk::GetPoolHeapAddress(), nns::atk::GetPoolHeapSize());
414 
415  // SoundArchivePlayer の初期化
416  std::size_t memSizeForSoundArchivePlayer = g_SoundArchivePlayer.GetRequiredMemSize( &g_SoundArchive );
417  g_pMemoryForSoundArchivePlayer = nns::atk::Allocate( memSizeForSoundArchivePlayer, nn::atk::SoundArchivePlayer::BufferAlignSize );
418  isSuccess = g_SoundArchivePlayer.Initialize(
419  &g_SoundArchive,
420  &g_SoundDataManager,
421  g_pMemoryForSoundArchivePlayer, memSizeForSoundArchivePlayer,
422  g_pMemoryForStreamBuffer, memSizeForStreamBuffer );
423  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundArchivePlayer" );
424 
425  g_CurrentEffectType = EffectType_None;
426  g_NextEffectType = EffectType_None;
427  g_IsEffectClearing = false;
428  g_IsEffectChanging = false;
429 }

431 void FinalizeAtk()
432 {
433  // エフェクトの解放処理
434  if ( !g_IsEffectClearing )
435  {
436  g_IsEffectClearing = PrepareEffectRemove( g_CurrentEffectType );
437  }
438  if ( g_IsEffectClearing )
439  {
440  // エフェクトが削除可能になるまで待つ
441  for ( ;; )
442  {
443  if ( CheckEffectRemovable( g_CurrentEffectType ) )
444  {
445  ClearEffect( g_CurrentEffectType );
446  g_IsEffectClearing = false;
447  break;
448  }
449  g_SoundArchivePlayer.Update();
450  }
451  }
452 
453  g_SoundArchivePlayer.Finalize();
454 
455  // 専用のヒープをメモリプールからデタッチ
456  nn::atk::SoundSystem::DetachMemoryPool(&g_MemoryPool);
457 
458  g_SoundDataManager.Finalize();
459  g_SoundArchive.Close();
460  g_SoundHeap.Destroy();
461  nn::atk::SoundSystem::Finalize();
462 
463  nns::atk::FreeForMemoryPool(g_pMemoryForStreamBuffer);
464  nns::atk::Free(g_pMemoryForSoundArchivePlayer);
465  nns::atk::Free(g_pMemoryForSoundDataManager);
466  nns::atk::Free(g_pMemoryForInfoBlock);
467  nns::atk::Free(g_pMemoryForSoundHeap);
468  nns::atk::Free(g_pMemoryForSoundSystem);
469 }

510 bool UpdateAtk()
511 {
512  // StartSound / Stop
513  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::A >() )
514  {
515  PlayWithStartSound(SEQ_MARIOKART, "SEQ_MARIOKART");
516  }
517  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::X >() )
518  {
519  PlayWithStartSound(SE_YOSHI, "SE_YOSHI");
520  }
521  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Y >() )
522  {
523  PlayWithStartSound(STRM_MARIOKART, "STRM_MARIOKART");
524  }
525  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::B >() )
526  {
527  g_SoundHandle.Stop( 0 );
528  }
529 
530  // Change EffectType
531  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Left >() )
532  {
533  if ( !g_IsEffectClearing )
534  {
535  g_IsEffectClearing = PrepareEffectRemove( g_CurrentEffectType );
536 
537  g_NextEffectType -= 1;
538  if ( g_NextEffectType < 0 )
539  {
540  g_NextEffectType = EffectType_Max;
541  }
542  g_IsEffectChanging = true;
543  }
544  }
545  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Right >() )
546  {
547  if ( !g_IsEffectClearing )
548  {
549  g_IsEffectClearing = PrepareEffectRemove( g_CurrentEffectType );
550 
551  g_NextEffectType += 1;
552  if ( g_NextEffectType > EffectType_Max )
553  {
554  g_NextEffectType = 0;
555  }
556  g_IsEffectChanging = true;
557  }
558  }
559 
560  // Print Usage
561  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::L >() )
562  {
563  PrintUsage();
564  }
565 
566  // Exit
567  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Start >() )
568  {
569  return false;
570  }
571 
572  g_SoundArchivePlayer.Update();
573 
574  // エフェクトが削除可能かをチェック
575  if ( g_IsEffectClearing )
576  {
577  if ( CheckEffectRemovable( g_CurrentEffectType ) )
578  {
579  ClearEffect( g_CurrentEffectType );
580  g_IsEffectClearing = false;
581  }
582  }
583 
584  // 次のエフェクトに変更する
585  if ( g_IsEffectChanging && !g_IsEffectClearing )
586  {
587  AppendEffect( g_NextEffectType );
588  g_CurrentEffectType = g_NextEffectType;
589  g_IsEffectChanging = false;
590  }
591 
592  return true;
593 }

サンプルプログラムの解説

先のサンプルプログラムの Atk に関連している処理の流れは以下の通りです(AtkSimple と違う部分には追加要素と書かれています)。

  • Atk ライブラリの初期化 (InitializeAtk())
    • AtkSimple で実行している初期化処理
  • SoundHeap にウェーブサウンドとシーケンスサウンドの再生に必要な情報をロード (LoadData())
  • メインループでの処理
    • キー操作による処理
      • サウンドの再生・停止処理を実行
        • エフェクト・メインのセンド量の設定(追加要素)
      • 適用するエフェクトの種類の変更(追加要素)
        • 適用するエフェクトの追加処理(AppendEffect())と削除処理(ClearEffect())(追加要素)
    • SoundArchivePlayer の更新
  • Atk ライブラリの終了処理 (FinalizeAtk())

キー操作を行うことで、サウンドに適用するエフェクトを変更することができます。 適用できるエフェクトはディレイ(左右チャンネル, 右チャンネルのみの 2 種類)およびリバーブ、 I3DL2 リバーブ、ユーザ定義のエフェクトの 5 種類となっています。

エフェクトを削除する場合は、まずエフェクトを無効化した後に(PrepareEffectRemove()) エフェクトが削除可能である事を確認した上で(CheckEffectRemovable())、削除を行う必要があります。
