nn::atk ライブラリのサウンドプロファイル機能のサンプルプログラムの解説です。

概要

Atk ライブラリでサウンドプロファイルを取得するサンプルです。

ファイル構成

本サンプルプログラムは Samples/Sources/Applications/AtkProfile 以下にあります。

必要な環境

とくになし

操作方法

コンソールに操作方法が表示されます。

注意事項

実行中のログが、画面上とコンソールに表示され、 起動直後のログとして操作方法が表示されます。

PC 版ではウィンドウは立ち上がらず、コンソール表示のみになる点にご注意ください。

実行手順

サンプルプログラムをビルドし、実行してください。

133 void InitializeAtk()
134 {
135  bool isSuccess = true;
136 
137  // SoundSystem の初期化
138  nn::atk::SoundSystem::SoundSystemParam param;
139  param.enableProfiler = true;
140  std::size_t memSizeForSoundSystem = nn::atk::SoundSystem::GetRequiredMemSize( param );
141  g_pMemoryForSoundSystem = nns::atk::Allocate( memSizeForSoundSystem, nn::atk::SoundSystem::WorkMemoryAlignSize );
142  isSuccess = nn::atk::SoundSystem::Initialize(
143  param,
144  reinterpret_cast<uintptr_t>( g_pMemoryForSoundSystem ),
145  memSizeForSoundSystem );
146  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundSystem" );
147 
148  // SoundHeap の初期化
149  g_pMemoryForSoundHeap = nns::atk::Allocate( SoundHeapSize );
150  isSuccess = g_SoundHeap.Create( g_pMemoryForSoundHeap, SoundHeapSize );
151  NN_ABORT_UNLESS( isSuccess, "cannot create SoundHeap" );
152 
153  // SoundArchive の初期化
154  const char* archiveAbsolutePath = nns::atk::GetAbsolutePath(ArchiveRelativePath);
155  isSuccess = g_SoundArchive.Open(archiveAbsolutePath);
156  NN_ABORT_UNLESS( isSuccess, "cannot open SoundArchive(%s)\n", archiveAbsolutePath );
157 
158  // SoundArchive のパラメータ情報をメモリにロード
159  std::size_t infoBlockSize = g_SoundArchive.GetHeaderSize();
160  g_pMemoryForInfoBlock = nns::atk::Allocate( infoBlockSize, nn::atk::FsSoundArchive::BufferAlignSize );
161  isSuccess = g_SoundArchive.LoadHeader( g_pMemoryForInfoBlock, infoBlockSize );
162  NN_ABORT_UNLESS( isSuccess, "cannot load InfoBlock" );
163 
164  // SoundDataManager の初期化
165  std::size_t memSizeForSoundDataManager = g_SoundDataManager.GetRequiredMemSize( &g_SoundArchive );
166  g_pMemoryForSoundDataManager = nns::atk::Allocate( memSizeForSoundDataManager, nn::atk::SoundDataManager::BufferAlignSize );
167  isSuccess = g_SoundDataManager.Initialize(
168  &g_SoundArchive,
169  g_pMemoryForSoundDataManager,
170  memSizeForSoundDataManager );
171  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundDataManager" );
172 
173  // SoundArchivePlayer で用いるストリームバッファの初期化
174  // ストリームバッファはメモリプール管理されているヒープから確保する必要があります。
175  std::size_t memSizeForStreamBuffer = g_SoundArchivePlayer.GetRequiredStreamBufferSize( &g_SoundArchive );
176  g_pMemoryForStreamBuffer = nns::atk::AllocateForMemoryPool(memSizeForStreamBuffer);
177 
178  // 専用のヒープをメモリプールにアタッチ
179  nn::atk::SoundSystem::AttachMemoryPool(&g_MemoryPool, nns::atk::GetPoolHeapAddress(), nns::atk::GetPoolHeapSize());
180 
181  // SoundArchivePlayer の初期化
182  std::size_t memSizeForSoundArchivePlayer = g_SoundArchivePlayer.GetRequiredMemSize( &g_SoundArchive );
183  g_pMemoryForSoundArchivePlayer = nns::atk::Allocate( memSizeForSoundArchivePlayer, nn::atk::SoundArchivePlayer::BufferAlignSize);
184  isSuccess = g_SoundArchivePlayer.Initialize(
185  &g_SoundArchive,
186  &g_SoundDataManager,
187  g_pMemoryForSoundArchivePlayer, memSizeForSoundArchivePlayer,
188  g_pMemoryForStreamBuffer, memSizeForStreamBuffer );
189  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundArchivePlayer" );
190 
191  nn::atk::SoundSystem::RegisterProfileReader(g_ProfileReader);
192 }

259 bool UpdateAtk()
260 {
261  // StartSound / Stop
262  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::A >() )
263  {
264  PlayWithStartSound(SEQ_MARIOKART, "SEQ_MARIOKART");
265  }
266  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::X >() )
267  {
268  PlayWithStartSound(SE_YOSHI, "SE_YOSHI");
269  }
270  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Y >() )
271  {
272  PlayWithStartSound(STRM_MARIOKART, "STRM_MARIOKART");
273  }
274  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::B >() )
275  {
276  g_SoundHandle.Stop( 0 );
277  }
278 
279  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Right >() )
280  {
281  PlayWithStartSound(STRM_MULTITRACK, "STRM_MULTITRACK");
282  }
283  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Up >() )
284  {
285  PlayWithStartSound(SE_YOSHI_PCM16, "SE_YOSHI_PCM16");
286  }
287  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Left >() )
288  {
289  PlayWithStartSound(STRM_MARIOKART_PCM16, "STRM_MARIOKART_PCM16");
290  }
291 
292  // プロファイル情報の取得/表示
293  bool isFirstProfile = false;
294  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Down >() )
295  {
296  g_IsReadingProfile = !g_IsReadingProfile;
297  isFirstProfile = g_IsReadingProfile;
298  NNS_LOG("---------------------------------------------------------------------------\n");
299  NNS_LOG("voices sndFr(us) ren(us) vo(us) mMix(us) fMix(us) sink(us) int(us)\n");
300  NNS_LOG("---------------------------------------------------------------------------\n");
301  }
302 
303  if (g_IsReadingProfile) {
304  int profileCount = g_ProfileReader.Read(g_SoundProfile, SoundProfileCountMax);
305  for( int i=0; i < profileCount; i++ )
306  {
307  nn::atk::SoundProfile profile = g_SoundProfile[i];
308 
309  NNS_LOG("%6d %9lld %7lld %6lld %8lld %8lld %8lld %7lld \n",
310  profile.totalVoiceCount,
311  profile.nwFrameProcess.GetSpan().GetMicroSeconds(),
312  profile.rendererFrameProcess.GetSpan().GetMicroSeconds(),
313  profile.voiceProcess.GetSpan().GetMicroSeconds(),
314  profile.mainMixProcess.GetSpan().GetMicroSeconds(),
315  profile.finalMixProcess.GetSpan().GetMicroSeconds(),
316  profile.sinkProcess.GetSpan().GetMicroSeconds(),
317  isFirstProfile ? 0 : profile.GetNwFrameProcessInterval(g_PrevNwFrameProcessBegin).GetMicroSeconds()
318  );
319 
320  g_PrevNwFrameProcessBegin = profile.nwFrameProcess.begin;
321  isFirstProfile = false;
322  }
323  }
324 
325  // Print Usage
326  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::L >() )
327  {
328  PrintUsage();
329  }
330 
331  // Exit
332  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Start >() )
333  {
334  return false;
335  }
336 
337  g_SoundArchivePlayer.Update();
338 
339  return true;
340 }

サンプルプログラムの解説

先のサンプルプログラムの Atk に関連している処理の流れは以下の通りです。

  • アロケータの初期化
    • 内部で２種類（メモリプール管理対象用とそれ以外）のアロケータを初期化。
  • Atk ライブラリの初期化 (InitializeAtk())
    • AtkSimple で実行している初期化処理
    • サウンドシステムのプロファイル機能の有効化(追加要素)
    • サウンドシステムにプロファイルリーダーを登録(追加要素)
  • SoundHeap にウェーブサウンドとシーケンスサウンドの再生に必要な情報をロード (LoadData())
  • メインループでの処理
    • キー操作による処理
      • サウンドの再生・停止処理を実行
      • プロファイルリーダーからサウンドプロファイル情報を読み込み(追加要素)
    • SoundArchivePlayer の更新
  • Atk ライブラリの終了処理 (FinalizeAtk())
    • サウンドシステムのプロファイルリーダーの登録を解除(追加要素)

サウンドシステムの初期化時にプロファイル機能を有効化し、プロファイルリーダーを登録する事で サウンドプロファイルを取得することができるようになります。 取得できる値につきましては、API リファレンスの nn::atk::SoundProfile の項目を参照ください。
