148 void* ReadResource( const char* filename )
149 {
150  nn::Result result;
151  nn::fs::FileHandle hFile;
152 
153  int64_t fileSize = 0;
154  result = nn::fs::OpenFile( &hFile, filename, nn::fs::OpenMode_Read );
155  NN_ASSERT( result.IsSuccess() );
156 
157  result = nn::fs::GetFileSize( &fileSize, hFile );
158  NN_ASSERT( result.IsSuccess() );
159 
160  nn::util::BinaryFileHeader fileHeader;
161  size_t readSize;
162  result = nn::fs::ReadFile( &readSize, hFile, 0, &fileHeader, sizeof( nn::util::BinaryFileHeader ) );
163  NN_ASSERT( result.IsSuccess() );
164  NN_ASSERT( readSize == sizeof( nn::util::BinaryFileHeader ) );
165  size_t alignment = fileHeader.GetAlignment();
166 
167  g_pMemory.AlignUp( alignment );
168  void* pBuffer = g_pMemory.Get();
169  result = nn::fs::ReadFile( &readSize, hFile, 0, pBuffer, static_cast< size_t >( fileSize ) );
170  NN_ASSERT( result.IsSuccess() );
171  NN_ASSERT( readSize == static_cast< size_t >( fileSize ) );
172  g_pMemory.Advance( static_cast< ptrdiff_t >( fileSize ) );
173 
174  nn::fs::CloseFile( hFile );
175 
176  return pBuffer;
177 }

968  const float height = 0.5f;

970

974  { 0, 1, 2, 3 }, // Front
975  { 3, 2, 6, 7 }, // Right
976  { 7, 6, 5, 4 }, // Back
977  { 4, 5, 1, 0 }, // Left
978  { 1, 5, 6, 2 }, // Top
979  { 4, 0, 3, 7 }, // Bottom
980  };
981 
982  // 頂点バッファを設定します。
983  const nn::util::Float3 VertexPos[8] =
984  {
985  NN_UTIL_FLOAT_3_INITIALIZER( width / 2.f, height, -depth / 2.f ),
986  NN_UTIL_FLOAT_3_INITIALIZER( width, -height, -depth ),
987  NN_UTIL_FLOAT_3_INITIALIZER( -width, -height, -depth ),
988  NN_UTIL_FLOAT_3_INITIALIZER( -width / 2.f, height, -depth / 2.f ),
989  NN_UTIL_FLOAT_3_INITIALIZER( width / 2.f, height, depth / 2.f ),
990  NN_UTIL_FLOAT_3_INITIALIZER( width, -height, depth ),
991  NN_UTIL_FLOAT_3_INITIALIZER( -width, -height, depth ),
992  NN_UTIL_FLOAT_3_INITIALIZER( -width / 2.f, height, depth / 2.f ),
993  };
994 
995  static const nn::util::Float2 VertexUv[4] =
996  {
997  NN_UTIL_FLOAT_2_INITIALIZER( 0.0f, 0.0f ),
998  NN_UTIL_FLOAT_2_INITIALIZER( 0.0f, 1.0f ),

1000  NN_UTIL_FLOAT_2_INITIALIZER( 1.0f, 0.0f ),

1004  {
1005  NN_UTIL_FLOAT_4_INITIALIZER( 1.0f, 0.2f, 0.2f, 1.0f ),
1006  NN_UTIL_FLOAT_4_INITIALIZER( 0.2f, 1.0f, 0.2f, 1.0f ),
1007  NN_UTIL_FLOAT_4_INITIALIZER( 0.2f, 0.2f, 1.0f, 1.0f ),
1008  NN_UTIL_FLOAT_4_INITIALIZER( 1.0f, 0.2f, 0.2f, 1.0f ),

1011  NN_UTIL_FLOAT_4_INITIALIZER( 1.0f, 0.2f, 0.2f, 1.0f ),
1012  NN_UTIL_FLOAT_4_INITIALIZER( 0.2f, 1.0f, 0.2f, 1.0f ),
1013  };
1014 
1015  const float xScale = 0.8f;
1016  const float yScale = 0.8f;
1017  const float zScale = 0.8f;
1018  uint32_t idxFace = 0;
1019  const int numIndices = 36;
1020  const int numVertices = 24;

1041  {

1044  for( idxFace = 0; idxFace < 6; idxFace++ )
1045  {
1046  pIndexData[idx++] = idxFace * 4 + 0;
1047  pIndexData[idx++] = idxFace * 4 + 1;
1048  pIndexData[idx++] = idxFace * 4 + 2;
1049 
1050  pIndexData[idx++] = idxFace * 4 + 0;
1051  pIndexData[idx++] = idxFace * 4 + 2;
1052  pIndexData[idx++] = idxFace * 4 + 3;

1054  }

1057  idxFace = 0;
1058 
1059  {
1060  nn::util::Float3* pPos = static_cast<nn::util::Float3*>( pMeshBuffer->GetVertexBufferCpuAddress(nns::gfx::PrimitiveRenderer::VertexAttribute_Pos));
1061

1064  for( idxFace = 0; idxFace < 6; idxFace++ )

1076  }

1083  idxCorner = 0;
1084  idxVertex = 0;
1085  for( idxFace = 0; idxFace < 6; idxFace++ )
1086  {
1087  for( idxCorner = 0; idxCorner < 4; idxCorner++ )
1088  {
1089  const size_t index = FaceTable[idxFace][idxCorner];
1090 
1091  // カラー

1093  {
1094  pColor[ idxVertex ].x = VertexColor[ index ].x;
1095  pColor[ idxVertex ].y = VertexColor[ index ].y;
1096  pColor[ idxVertex ].z = VertexColor[ index ].z;
1097  pColor[ idxVertex ].w = VertexColor[ index ].w;
1098  }
1099 
1100  // UV

1102  {
1103  pUv[ idxVertex ].x = VertexUv[ idxCorner ].x;
1104  pUv[ idxVertex ].y = VertexUv[ idxCorner ].y;
1105  }
1106  idxVertex++;
1107  }
1108  }
1109  }
1110  }

サンプルプログラムの解説

サンプルプログラムの処理の流れは以下の通りです。

  • デバイスを初期化
  • メモリプールを初期化
  • スワップチェーンを初期化
  • キューを初期化
  • コマンドバッファを初期化
  • ビューポートを初期化
  • ラスタライザステートを初期化
  • カラーバッファを初期化
  • 深度ステンシルバッファを初期化
  • UserShader向けバッファを初期化
  • サンプラを初期化
  • サンプラデスクリプタプールを初期化
  • テクスチャデスクリプタプールを初期化
  • テクスチャリソースを初期化
  • nns::gfx::PrimitiveRendererを初期化
  • UserShaderを初期化
  • ループ開始
  • シェーダバリエーションを選択
  • コマンドリストを作成
  • コマンドリストを実行
  • ディスプレイへプレゼンテーション
  • ループ開始に戻る
  • 各種オブジェクトを破棄
