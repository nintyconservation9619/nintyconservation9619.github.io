This tutorial builds off of Tutorial07 by adding functionality to display GPU performance counters on the screen.

Overview

This tutorial builds off of Tutorial07 by adding functionality to display GPU performance counters on the screen. The performance counters are queried by using the StatTracker helper and are printed to the screen with the DebugTextRenderer. More details on the GPU statistics can be found in StatTracker.cpp/.h.

Expected Output

File Structure

The main tutorial file and Visual Studio solutions can be found at Samples/Sources/Applications/NvnTutorial08GpuStatistics

The tutorial library which contains common code shared by the tutorials can be found at Samples/Sources/Libraries/NvnTutorial

System Requirements

No extra system requirements.

Operation Procedure

This sample runs on its own without any additional input. Hitting escape on Windows or touching the touch screen on NX will exit the program.

Precautions

None.

Execution Procedure

Build the Visual Solution in the desired configuration and run it.

463 /*
464  * GpuStatistics::UpdateRenderTargets
465  * ----------------------------------
466  * Gets the index of the current render target from the NVNwindow
467  * and records a command buffer that sets it up to be rendered to
468  * and clears it.
469  */
470 int GpuStatistics::UpdateRenderTargets()
471 {
472  /* Get next render target to be used */
473  NVNwindowAcquireTextureResult result = nvnWindowAcquireTexture(m_pWindow, &m_WindowSync, &m_CurrentWindowIndex);
474 
475  NN_ASSERT(result == NVN_WINDOW_ACQUIRE_TEXTURE_RESULT_SUCCESS);
476 
477  /* Record the command buffer to set and clear the target. */
478  m_pRenderTargetManagedCommandBuffer->BeginRecording();
479  {
480  NVNcommandBuffer* commandBuffer = m_pRenderTargetManagedCommandBuffer->GetCommandBuffer();
481  nvnCommandBufferSetRenderTargets(commandBuffer, 1, &m_RenderTargets[m_CurrentWindowIndex], NULL, m_pDepthTarget, NULL);
482 
483  nvnCommandBufferSetScissor(commandBuffer, 0, 0, m_ScreenWidth, m_ScreenHeight);
484  nvnCommandBufferSetViewport(commandBuffer, 0, 0, m_ScreenWidth, m_ScreenHeight);
485 
486  float clear_color[4] = { 0.4f, 0.55f, 0.6f, 1.0f };
487  nvnCommandBufferClearColor(commandBuffer, 0, clear_color, NVN_CLEAR_COLOR_MASK_RGBA);
488  nvnCommandBufferClearDepthStencil(commandBuffer, 1.0, NVN_TRUE, 0, 0);
489  }
490  m_RenderTargetCommandHandle = m_pRenderTargetManagedCommandBuffer->EndRecording();
491 
492  /* Return the index. */
493  return m_CurrentWindowIndex;
494 }

609 /*
610  * GpuStatistics::Draw
611  * -------------------
612  * This function runs the threads in the render thread
613  * pool to generate the command buffers to draw the list
614  * of objects and submits them to the queue. It also presents
615  * the current render target to the screen. Additionally,
616  * this function uses the StatTracker helper class to display
617  * GPU performance counters to the screen.
618  */
619 void GpuStatistics::Draw(uint64_t millisec)
620 {
621  NN_UNUSED(millisec);
622  static uint32_t s_Frame = 0;
623  ++s_Frame;
624 
625  /*
626  * Reset the counter values. This causes the GPU counters to be reset for the
627  * current frame. If this is not called, the values will be accumulated until
628  * reset is called.
629  */
630  m_pStatTracker->ResetCounters(&m_Queue);
631 
632  /* Get the current render target and setup a command buffer to set it. */
633  int index = UpdateRenderTargets();
634 
635  /* Update the transform data for the objects to be drawn. */
636  UpdateObjectData(s_Frame);
637 
638  /*
639  * Run the threads in the thread pool to generate
640  * command buffers to render the objects in the work
641  * list.
642  */
643  std::vector<NVNcommandHandle> outHandles;
644  m_pRenderThreadPool->Run(&outHandles, &m_Objects);
645  m_pRenderThreadPool->Wait();
646  m_pRenderThreadPool->SwapCommandMemory();
647 
648  /* Insert a fence in the sync manager. */
649  m_pFrameBufferedSyncManager->InsertFence();
650 
655  nvnSyncWait(&m_WindowSync, NVN_WAIT_TIMEOUT_MAXIMUM);
656 
657  /* Submit command buffer for switching render targets */
658  nvnQueueSubmitCommands(&m_Queue, 1, &m_RenderTargetCommandHandle);
659 
660  /* Submit the command buffers generated by the thread pool. */
661  nvnQueueSubmitCommands(&m_Queue, static_cast<int>(outHandles.size()), &outHandles[0]);
662 
663  /* Copy the counters to the CPU through the command buffer */
664  m_pStatTracker->GetCounters(&m_Queue);
665 
666  /*
667  * Reports the current values of the GPU counters. The placement of reset
668  * and report in this application is such that only the rendered models get
669  * counted in the statistics while the text rendering is left out.
670  */
671  m_pStatTracker->ReportCounters(&m_Queue);
672 
673  /* Print the counters to the screen with the text renderer. */
674  m_pStatTracker->PrintCounters(m_pDebugTextRenderer, 0, 0);
675 
676  /* Render the text. */
677  m_pDebugTextRenderer->Draw(m_RenderTargets[index], m_ScreenWidth, m_ScreenHeight);
678 
679  /* Present the render target to the screen. */
680  nvnQueuePresentTexture(&m_Queue, m_pWindow, index);
681 
682  /* Swap memory for command buffer and uniform buffers */
683  m_pFrameBufferedSyncManager->SwapPools();
684 }

696  m_ScreenHeight = height;

698  /* Check for the window being minimized or having no visible surface. */
699  if (width == 0 || height == 0)
700  {
701  return;
702  }
703 
704  /* If it's the first time Resize is called, allocate the NVNwindow. */
705  if (m_pWindow == NULL)
706  {
707  m_pWindow = new NVNwindow;
708  }
709  /*
710  * Otherwise finalize (free) the NVNwindow used for the previous window size.
711  * The NVNWindow must be finalized before a render target it owns is finalized.
712  */
713  else
714  {
715  nvnWindowFinalize(m_pWindow);
716  }
717 
718  /* Set up the builder for the render target. */
719  nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
720  nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_DISPLAY_BIT | NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
721  nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
722  nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
723  nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_RGBA8);
724 
725  for(int i = 0; i < g_NumColorBuffers; ++i)
726  {
727  /* If it's the first time Resize is called, allocate the render target. */
728  if(m_RenderTargets[i] == NULL)
729  {
730  m_RenderTargets[i] = new NVNtexture;
731  }
732  /* Otherwise finalize (free) the render target used for the previous window size. */
733  else
734  {
735  nvnTextureFinalize(m_RenderTargets[i]);
736  }
737 
738  nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * i);
739 
740  /* Create the texture using the current state of the texture builder. */
741  nvnTextureInitialize(m_RenderTargets[i], &m_RenderTargetBuilder);
742  }
743 
744  if (m_pDepthTarget == NULL)
745  {
746  m_pDepthTarget = new NVNtexture;
747  }
748  else
749  {
750  nvnTextureFinalize(m_pDepthTarget);
751  }
752 
753  /* Initialize depth buffer for render target. */
754  nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
755  nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
756  nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
757  nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
758  nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_DEPTH32F);
759  nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * g_NumColorBuffers);
760  nvnTextureInitialize(m_pDepthTarget, &m_RenderTargetBuilder);
761 
762  /* Pass off the render targets to the window. */
763  nvnWindowBuilderSetTextures(&m_WindowBuilder, g_NumColorBuffers, m_RenderTargets);
764  nvnWindowInitialize(m_pWindow, &m_WindowBuilder);
765 
766  /* Remake the projection matrix with the new screen size. */
767  nn::util::Matrix4x4fType projMat44;
768  nn::util::MatrixPerspectiveFieldOfViewRightHanded(&projMat44, 60.0f * 3.14159f / 180.0f, static_cast<float>(m_ScreenWidth) / static_cast<float>(m_ScreenHeight), 1.0f, 1000.0f);
769 
770  nn::util::Matrix4x4fType camMat44;
771  nn::util::MatrixIdentity(&camMat44);
772  for (size_t i = 0; i < m_Objects.size(); ++i)
773  {
774  reinterpret_cast<GraphicsObjectCube*>(m_Objects[i])->SetProjectionMatrix(projMat44);
775  reinterpret_cast<GraphicsObjectCube*>(m_Objects[i])->SetCameraMatrix(camMat44);
776  }
777 }

This tutorial renders a 4x4x4 grid of cubes that alternate between opaque and wire frame. Each cube is rendered by one of four render threads. It also displays a list of GPU statistics on screen using the debug text renderer.
