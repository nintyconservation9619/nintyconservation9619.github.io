This sample demonstrates basic initialization of OpenGL/OpenGL ES and renders a single triangle.

Overview

This sample demonstrates basic initialization of OpenGL and renders a single triangle. This is acccomplished through the use of OpenGL or ES version of it. Define the USE_OPENGL_ES macro in GraphicsHelper.h in order to target the OpenGL ES 3.2 profile as opposed to OpenGL 4.5 in this sample.

Expected Output

File Structure

The main sample file and Visual Studio solutions can be found at Samples/Sources/Applications/GlSimple

System Requirements

No extra system requirements.

Operation Procedure

This sample runs on its own without any additional input. Touching the touch screen on NX will exit the program.

Precautions

None.

Execution Procedure

Build the Visual Solution in the desired configuration and run it.

144 void InitializeShaders()
145 {
146  GLint result;
147  GLchar shaderLog[1024];
148  GLsizei shaderLogSize;
149 
150  s_VertexShaderId = glCreateShader(GL_VERTEX_SHADER);
151  NN_ASSERT(s_VertexShaderId != 0, "Failed to create vertex shader\n");
152  glShaderSource(s_VertexShaderId, 1, &VertexShaderSource, 0);
153  glCompileShader(s_VertexShaderId);
154  glGetShaderiv(s_VertexShaderId, GL_COMPILE_STATUS, &result);
155  if (!result)
156  {
157  glGetShaderInfoLog(s_VertexShaderId, sizeof(shaderLog), &shaderLogSize, shaderLog);
158  NN_ASSERT(false, "Failed to compile vertex shader: %s\n", shaderLog);
159  }
160 
161  s_FragmentShaderId = glCreateShader(GL_FRAGMENT_SHADER);
162  NN_ASSERT(s_FragmentShaderId != 0, "Failed to create fragment shader\n");
163  glShaderSource(s_FragmentShaderId, 1, &FragmentShaderSource, 0);
164  glCompileShader(s_FragmentShaderId);
165  glGetShaderiv(s_FragmentShaderId, GL_COMPILE_STATUS, &result);
166  if (!result)
167  {
168  glGetShaderInfoLog(s_FragmentShaderId, sizeof(shaderLog), &shaderLogSize, shaderLog);
169  NN_ASSERT(false, "Failed to compile fragment shader: %s\n", shaderLog);

171 
172  s_ShaderProgramId = glCreateProgram();

175  glAttachShader(s_ShaderProgramId, s_VertexShaderId);
