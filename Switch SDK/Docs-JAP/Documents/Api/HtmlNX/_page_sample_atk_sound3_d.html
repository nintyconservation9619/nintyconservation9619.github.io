nn::atk ライブラリの Sound3D に関する機能のサンプルプログラムの解説です。

概要

Atk ライブラリの Sound3D を用いて音声を再生するサンプルです。 ドップラー効果や 3D Pan などの立体的な空間での音響を再現します。

ファイル構成

本サンプルプログラムは Samples/Sources/Applications/AtkSound3D 以下にあります。

必要な環境

とくになし

操作方法

コンソールに操作方法が表示されます。

注意事項

実行中のログが、画面上とコンソールに表示され、 起動直後のログとして操作方法が表示されます。

PC 版ではウィンドウは立ち上がらず、コンソール表示のみになる点にご注意ください。

実行手順

サンプルプログラムをビルドし、実行してください。

144 void CalcListenerMatrix( nn::util::Matrix4x3fType* mtx )
145 {
146  // リスナーの位置
147  nn::util::Vector3fType pos;
148  nn::util::VectorSet( &pos, 0.0f, 0.0f, -3.0f );
149 
150  // リスナーの Up 方向のベクトル
151  nn::util::Vector3fType upVec;
152  nn::util::VectorSet( &upVec, 0.0f, 1.0f, 0.0f );
153 
154  // リスナーの方向ベクトル
155  nn::util::Vector3fType direction;
156  float directionSin;
157  float directionCos;
158  nn::util::SinCosEst( &directionSin, &directionCos, 0.0f );
159  nn::util::VectorSet( &direction, -directionSin, 0.0f, -directionCos );
160 
161  // リスナー行列生成
162  nn::util::Vector3fType target;
163  nn::util::VectorAdd( &target, pos, direction );
164  nn::util::MatrixLookAtRightHanded( mtx, pos, target, upVec );
165 }

167 void InitializeAtk()
168 {
169  bool isSuccess = true;
170 
171  // SoundSystem の初期化
172  nn::atk::SoundSystem::SoundSystemParam param;
173  std::size_t memSizeForSoundSystem = nn::atk::SoundSystem::GetRequiredMemSize( param );
174  g_pMemoryForSoundSystem = nns::atk::Allocate( memSizeForSoundSystem, nn::atk::SoundSystem::WorkMemoryAlignSize );
175  isSuccess = nn::atk::SoundSystem::Initialize(
176  param,
177  reinterpret_cast<uintptr_t>( g_pMemoryForSoundSystem ),
178  memSizeForSoundSystem );
179  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundSystem" );
180 
181  // SoundHeap の初期化
182  g_pMemoryForSoundHeap = nns::atk::Allocate( SoundHeapSize );
183  isSuccess = g_SoundHeap.Create( g_pMemoryForSoundHeap, SoundHeapSize );
184  NN_ABORT_UNLESS( isSuccess, "cannot create SoundHeap" );
185 
186  // SoundArchive の初期化
187  const char* archiveAbsolutePath = nns::atk::GetAbsolutePath(ArchiveRelativePath);
188  isSuccess = g_SoundArchive.Open(archiveAbsolutePath);
189  NN_ABORT_UNLESS( isSuccess, "cannot open SoundArchive(%s)\n", archiveAbsolutePath );
190 
191  // SoundArchive のパラメータ情報をメモリにロード
192  std::size_t infoBlockSize = g_SoundArchive.GetHeaderSize();
193  g_pMemoryForInfoBlock = nns::atk::Allocate( infoBlockSize, nn::atk::FsSoundArchive::BufferAlignSize );
194  isSuccess = g_SoundArchive.LoadHeader( g_pMemoryForInfoBlock, infoBlockSize );
195  NN_ABORT_UNLESS( isSuccess, "cannot load InfoBlock" );
196 
197  // SoundDataManager の初期化
198  std::size_t memSizeForSoundDataManager = g_SoundDataManager.GetRequiredMemSize( &g_SoundArchive );
199  g_pMemoryForSoundDataManager = nns::atk::Allocate( memSizeForSoundDataManager, nn::atk::SoundDataManager::BufferAlignSize );
200  isSuccess = g_SoundDataManager.Initialize(
201  &g_SoundArchive,
202  g_pMemoryForSoundDataManager,
203  memSizeForSoundDataManager );
204  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundDataManager" );
205 
206  // Sound3DManager の初期化
207  std::size_t memSizeForSound3DManager = g_Sound3DManager.GetRequiredMemSize( &g_SoundArchive );
208  g_pMemoryForSound3DManager = nns::atk::Allocate( memSizeForSound3DManager );
209  isSuccess = g_Sound3DManager.Initialize(
210  &g_SoundArchive,
211  g_pMemoryForSound3DManager,
212  memSizeForSound3DManager );
213  NN_ABORT_UNLESS( isSuccess, "cannot initialize Sound3DManager" );
214 
215  g_FilterType= nn::atk::BiquadFilterType_None;
216  g_Sound3DManager.SetBiquadFilterType( g_FilterType );
217  g_Sound3DManager.SetMaxPriorityReduction( 32 );
218  g_Sound3DManager.SetSonicVelocity( 340.0f / 60.0f );
219 
220  // SoundArchivePlayer で用いるストリームバッファの初期化
221  // ストリームバッファはメモリプール管理されているヒープから確保する必要があります。
222  std::size_t memSizeForStreamBuffer = g_SoundArchivePlayer.GetRequiredStreamBufferSize( &g_SoundArchive );
223  g_pMemoryForStreamBuffer = nns::atk::AllocateForMemoryPool(memSizeForStreamBuffer);
224 
225  // 専用のヒープをメモリプールにアタッチ
226  nn::atk::SoundSystem::AttachMemoryPool(&g_MemoryPool, nns::atk::GetPoolHeapAddress(), nns::atk::GetPoolHeapSize());
227 
228  // SoundArchivePlayer の初期化
229  std::size_t memSizeForSoundArchivePlayer = g_SoundArchivePlayer.GetRequiredMemSize( &g_SoundArchive );
230  g_pMemoryForSoundArchivePlayer = nns::atk::Allocate( memSizeForSoundArchivePlayer, nn::atk::SoundArchivePlayer::BufferAlignSize);
231  isSuccess = g_SoundArchivePlayer.Initialize(
232  &g_SoundArchive,
233  &g_SoundDataManager,
234  g_pMemoryForSoundArchivePlayer, memSizeForSoundArchivePlayer,
235  g_pMemoryForStreamBuffer, memSizeForStreamBuffer );
236  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundArchivePlayer" );
237 
238  // Sound3DListener の初期化
239  nn::util::Matrix4x3fType listenerMtx;
240  nn::util::Vector3fType zeroVelocity;
241  CalcListenerMatrix( &listenerMtx );
242  nn::util::VectorSet( &zeroVelocity, 0.0f, 0.0f, 0.0f );
243 
244  g_Sound3DListener.SetMatrix( listenerMtx );
245  g_Sound3DListener.SetVelocity( zeroVelocity );
246  g_Sound3DListener.SetMaxVolumeDistance( 5.0f );
247  g_Sound3DListener.SetUnitDistance( 5.0f );
248  g_Sound3DListener.SetInteriorSize( 5.0f );
249  g_Sound3DManager.AddListener( &g_Sound3DListener );
250 
251  // Sound3DActor の初期化
252  g_Sound3DActor.Initialize( &g_SoundArchivePlayer, &g_Sound3DManager );
253  nn::util::VectorSet( &g_ActorPos, 0.0f, 0.0f, 0.0f );
254  g_Sound3DActor.SetPosition( g_ActorPos );
255  g_Sound3DActor.SetVelocity( zeroVelocity );
256 }

390 bool UpdateAtk()
391 {
392  if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::R >() )
393  {
394  // FilterType の変更
395  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::A >() )
396  {
397  ChangeFilterType();
398  }
399 
400  // StopSound
401  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::B >() )
402  {
403  StopSound();
404  }
405 
406  // Sound3DActor の座標をリセット
407  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::X >() )
408  {
409  ResetActorPosition();
410  }
411  }
412  else
413  {
414  // PlaySound
415  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::A >() )
416  {
417  PlaySound3D( SE_IDLE_DOPPLER_FILTER, "SE_IDLE_DOPPLER_FILTER");
418  }
419  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::B >() )
420  {
421  PlaySound3D( SE_IDLE_DOPPLER, "SE_IDLE_DOPPLER");
422  }
423  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::X >() )
424  {
425  PlaySound3D( SE_IDLE_FILTER, "SE_IDLE_FILTER");
426  }
427  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Y >() )
428  {
429  PlaySound3D( SE_IDLE, "SE_IDLE");
430  }
431  }
432 
433  // Sound3DActor の座標変更
434  if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::Left >() )
435  {
436  MoveActor( -ControlPadStep, 0.0f, 0.0f );
437  }
438  if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::Right >() )
439  {
440  MoveActor( ControlPadStep, 0.0f, 0.0f );
441  }
442  if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::Up >() )
443  {
444  MoveActor( 0.0f, 0.0f, -ControlPadStep );
445  }
446  if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::Down >() )
447  {
448  MoveActor( 0.0f, 0.0f, ControlPadStep );
449  }
450 
451  // Print Usage
452  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::L >() )
453  {
454  PrintUsage();
455  }
456 
457  // Exit
458  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Start >() )
459  {
460  return false;
461  }
462 
463  g_Sound3DActor.SetPosition( g_ActorPos );
464  g_SoundArchivePlayer.Update();
465 
466  return true;
467 }

サンプルプログラムの解説

先のサンプルプログラムの Atk に関連している処理の流れは以下の通りです（AtkSimple と違う部分には追加要素と書かれています）。

  • Atk ライブラリの初期化 (InitializeAtk())
    • SoundArchivePlayer を除く AtkSimple で実行している初期化処理
    • Sound3DManager の初期化（追加要素）
    • SoundArchivePlayer の初期化
    • Sound3DListener の初期化（追加要素）
    • Sound3DActor の初期化（追加要素）
  • SoundHeap にウェーブサウンドの再生に必要な情報をロード (LoadData())
  • メインループでの処理
    • キー操作による処理
      • サウンドの再生・停止処理を実行
      • 適用する 3D フィルターの種類を変更（追加要素）
      • Sound3DActor の移動（追加要素）
    • Sound3DActor の位置を更新（追加要素）
    • SoundArchivePlayer の更新
  • Atk ライブラリの終了処理 (FinalizeAtk())

キー操作を行うことで、Sound3DActor を移動させることができます。 Sound3DActor を移動させることで、立体的な音響の変化を確認することができます。 立体的な音響の変化は SoundMaker で設定されたパラメータによって調整することができます。 このサンプルではドップラー効果の ON/OFF, フィルター処理の ON/OFF をそれぞれ設定した 4 つのサウンドを再生することができます。
