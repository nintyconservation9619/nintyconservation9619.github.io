アカウントライブラリの基本的な機能について、アプリケーションから使用する方法を示します。

概要

アカウントライブラリの基本的な機能について、アプリケーションから使用する方法を示します。 アカウントライブラリの基本的な機能には次が含まれます。

  • ユーザーアカウントの選択機能
  • ユーザーアカウントの属性を取得, 変更する機能
  • ユーザーアカウントの Open, Close 機能
  • ユーザーのネットワークサービスアカウントを使用する機能

ファイル構成

本サンプルプログラムは Samples/Sources/Applications/AccountApplication 以下にあります。

必要な環境

本サンプルプログラムは開発機向けにのみビルドし実行することができます。 Windows 環境をサポートしていないことに注意してください。

操作方法

一般的なサンプルプログラムと同様に本プログラムをビルドし、実行してください。 このプログラムの実行の経過は画面への文字の描画、及び実行ログとして出力されます。

解説

このサンプルプログラムは次の順序で処理を行います。

  • ユーザーアカウントの選択
  • ユーザーアカウントの属性の取得
  • ユーザーアカウントの Open
  • ユーザーアカウントの属性の変更
  • ネットワークサービスアカウントの使用
  • ユーザーアカウントの Close

104  for (auto t = 0; t < Width; ++ t)
105  {
106  const uint8_t G = static_cast<char>(((t + 1) * 255) / Width);
107  const uint8_t B = 0;
108  const uint8_t A = 255;
109 
110  uint8_t* pixel = reinterpret_cast<uint8_t*>(buffer) + ((s * Width + t) * 4);
111  pixel[0] = R;
112  pixel[1] = G;
113  pixel[2] = B;
114  pixel[3] = A;

188 // ネットワークサービスアカウントの ID トークンの取得
189 Program::Error Program::AcquireNsaIdToken(size_t* pOutSize, char* buffer, size_t bufferSize, const nn::account::UserHandle& handle) NN_NOEXCEPT
190 {
191  // ネットワークサービスアカウントの確定
192  // - ユーザー操作によるキャンセル時には nn::account::ResultCancelledByUser が返る。
193  auto r = nn::account::EnsureNetworkServiceAccountAvailable(handle);
194  if (!r.IsSuccess())
195  {
196  if (nn::account::ResultCancelledByUser::Includes(r))
197  {
198  // ユーザーによるキャンセル処理
199  return Error_Cancel;
200  }
201  NN_ABORT_UNLESS_RESULT_SUCCESS(r);
202  }
203 
204  // ネットワークサービスアカウントの情報
205  // - ユーザーはネットワークサービスアカウントを付け替えることができるため、アプリケーション起動ごとに変わっている可能性がある。
206  nn::account::NetworkServiceAccountId nsaId;
207  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::account::GetNetworkServiceAccountId(&nsaId, handle));
208  Printfln("- Network Service Account ID: %016llx", nsaId.id);
209 
210  // ネットワークサービスアカウントの ID トークンキャッシュを確定
211  nn::account::AsyncContext ctx;
212  auto durationMsec = -1ll;
213  {
214  // このスコープの処理時間を測る
215  auto begin = nn::os::GetSystemTick();
216  NN_UTIL_SCOPE_EXIT
217  {
218  auto end = nn::os::GetSystemTick();
219  durationMsec = (end - begin).ToTimeSpan().GetMilliSeconds();
220  };
221 
222  // ネットワークサービスアカウントへのログイン + ID トークンのキャッシュ API
223  // - すでにキャッシュが存在する場合は即時完了する。
224  // - ctx.Cancel() を呼ぶと、可能な限り早くイベントがシグナルされる。
225  NN_ABORT_UNLESS_RESULT_SUCCESS(nn::account::EnsureNetworkServiceAccountIdTokenCacheAsync(&ctx, handle));
226  nn::os::SystemEvent e;
227  NN_ABORT_UNLESS_RESULT_SUCCESS(ctx.GetSystemEvent(&e));
228  e.Wait();
229  }
230  r = ctx.GetResult();
231  if (!r.IsSuccess())
232  {
233  if (nn::account::ResultNetworkServiceAccountUnavailable::Includes(r))
234  {
235  // ログイン時にネットワークサービスアカウントが利用できないことが発覚
236  // - nn::account::EnsureNetworkServiceAccountAvailable() でエラーを解消できるため、再試行する。
237  // - 再試行や本体システム UI の表示が困難な場合は nn::error::ShowError() に渡してもよい。
238  return Error_Retry;
239  }
240  else if (nn::account::ResultNetworkCommunicationError::Includes(r))
241  {
242  // ログイン時に通信関係のエラーが発生
243  // - これにはアプリケーションの更新が必要な場合や、本体更新が必要な場合も含まれる。
244  // - nn::error::ShowError() に渡し、詳細なエラー内容をユーザーに提示する必要がある。
245  nn::err::ShowError(r);
246  return Error_Cancel;
247  }
248  // ctx.Cancel() が呼ばれていると、ここで nn::account::ResultCancelled() が返る場合がある。
249  // しかしこのサンプルでは ctx.Cancel() を呼んでいないため、これが返ることはない。
250  NN_ABORT_UNLESS_RESULT_SUCCESS(r);
251  }
252  Printfln("- Network Service Account login turnaround: %lld msec", durationMsec);
253 
254  // ネットワークサービスアカウントの ID トークンキャッシュを取得する
255  size_t idTokenLength;
256  r = nn::account::LoadNetworkServiceAccountIdTokenCache(&idTokenLength, buffer, bufferSize, handle);
257  if (!r.IsSuccess())
258  {
259  if (false
260  || nn::account::ResultNetworkServiceAccountUnavailable::Includes(r)
261  || nn::account::ResultTokenCacheUnavailable::Includes(r))
262  {
263  // ネットワークサービスアカウントへのログイン後にアプリケーションがフォーカスを失い...
264  // - その間に NSA が無効になった場合
265  // - その間に ID トークンキャッシュが無効になった場合
266  return Error_Retry;
267  }
268  // NSA 利用不可, トークンキャッシュ利用不可以外のエラーが返ることは通常ない。
269  NN_ABORT_UNLESS_RESULT_SUCCESS(r);
270  }
271  Printfln("- Network Service Account ID token: %.*s", idTokenLength, buffer);
272 
273  *pOutSize = idTokenLength;
274  return Error_Success;
275 }

277 // シングルユーザーのオンラインモードを想定した処理
278 void Program::SceneSingleUserWithNetwork(const nn::account::UserHandle& handle) NN_NOEXCEPT
279 {
280  // ネットワークサービスアカウントの ID トークンを取得する
281  auto* idToken = reinterpret_cast<char*>(std::malloc(nn::account::NetworkServiceAccountIdTokenLengthMax));
282  NN_UTIL_SCOPE_EXIT
283  {
284  std::free(idToken);
285  };
286  size_t idTokenLength;
287 
288  bool success = false;
289  while (!success)
290  {
291  // ネットワーク接続要求
292  while (NN_STATIC_CONDITION(true))
293  {
294  nn::nifm::SubmitNetworkRequestAndWait();
295  if (!nn::nifm::IsNetworkAvailable())
296  {
297  auto r = nn::nifm::HandleNetworkRequestResult();
298  if (!r.IsSuccess())
299  {
300  if (!nn::nifm::ResultErrorHandlingCompleted::Includes(r))
301  {
302  // エラーかつ、それを解消できなかった場合はシーンを終了する。
303  return;
304  }
305  // エラーかつ、それを解消できた場合はリトライする。
306  continue;
307  }
308  // break
309  }
310  // ネットワーク接続を利用できる。
311  break;
312  }
313 
314  auto error = AcquireNsaIdToken(&idTokenLength, idToken, nn::account::NetworkServiceAccountIdTokenLengthMax, handle);
315  switch (error)
316  {
317  case Error_Success:
318  success = true;
319  Printfln("-> Success");
320  break;
321  case Error_Cancel:
322  Printfln("-> Cancelled");
323  return;
324  case Error_Retry:
325  Printfln("-> Retry required");
326  break;
327  default:
328  NN_UNEXPECTED_DEFAULT;
329  }
330  }
