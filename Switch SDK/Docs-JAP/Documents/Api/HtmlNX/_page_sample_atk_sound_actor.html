nn::atk ライブラリのサウンドアクター機能に関するサンプルプログラムの解説です。

概要

Atk ライブラリのサウンドアクター機能を用いてサウンドを再生するサンプルです。

ファイル構成

本サンプルプログラムは Samples/Sources/Applications/AtkSoundActor 以下にあります。

必要な環境

とくになし

操作方法

コンソールに操作方法が表示されます。

注意事項

実行中のログが、画面上とコンソールに表示され、 起動直後のログとして操作方法が表示されます。

PC 版ではウィンドウは立ち上がらず、コンソール表示のみになる点にご注意ください。

実行手順

サンプルプログラムをビルドし、実行してください。

146 void InitializeAtk()
147 {
148  bool isSuccess = true;
149 
150  // SoundSystem の初期化
151  nn::atk::SoundSystem::SoundSystemParam param;
152  std::size_t memSizeForSoundSystem = nn::atk::SoundSystem::GetRequiredMemSize( param );
153  g_pMemoryForSoundSystem = nns::atk::Allocate( memSizeForSoundSystem, nn::atk::SoundSystem::WorkMemoryAlignSize );
154  isSuccess = nn::atk::SoundSystem::Initialize(
155  param,
156  reinterpret_cast<uintptr_t>( g_pMemoryForSoundSystem ),
157  memSizeForSoundSystem );
158  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundSystem" );
159 
160  // SoundHeap の初期化
161  g_pMemoryForSoundHeap = nns::atk::Allocate( SoundHeapSize );
162  isSuccess = g_SoundHeap.Create( g_pMemoryForSoundHeap, SoundHeapSize );
163  NN_ABORT_UNLESS( isSuccess, "cannot create SoundHeap" );
164 
165  // SoundArchive の初期化
166  const char* archiveAbsolutePath = nns::atk::GetAbsolutePath(ArchiveRelativePath);
167  isSuccess = g_SoundArchive.Open(archiveAbsolutePath);
168  NN_ABORT_UNLESS( isSuccess, "cannot open SoundArchive(%s)\n", archiveAbsolutePath );
169 
170  // SoundArchive のパラメータ情報をメモリにロード
171  std::size_t infoBlockSize = g_SoundArchive.GetHeaderSize();
172  g_pMemoryForInfoBlock = nns::atk::Allocate( infoBlockSize, nn::atk::FsSoundArchive::BufferAlignSize );
173  isSuccess = g_SoundArchive.LoadHeader( g_pMemoryForInfoBlock, infoBlockSize );
174  NN_ABORT_UNLESS( isSuccess, "cannot load InfoBlock" );
175 
176  // SoundDataManager の初期化
177  std::size_t memSizeForSoundDataManager = g_SoundDataManager.GetRequiredMemSize( &g_SoundArchive );
178  g_pMemoryForSoundDataManager = nns::atk::Allocate( memSizeForSoundDataManager, nn::atk::SoundDataManager::BufferAlignSize );
179  isSuccess = g_SoundDataManager.Initialize(
180  &g_SoundArchive,
181  g_pMemoryForSoundDataManager,
182  memSizeForSoundDataManager );
183  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundDataManager" );
184 
185  // SoundArchivePlayer で用いるストリームバッファの初期化
186  // ストリームバッファはメモリプール管理されているヒープから確保する必要があります。
187  std::size_t memSizeForStreamBuffer = g_SoundArchivePlayer.GetRequiredStreamBufferSize( &g_SoundArchive );
188  g_pMemoryForStreamBuffer = nns::atk::AllocateForMemoryPool(memSizeForStreamBuffer);
189 
190  // 専用のヒープをメモリプールにアタッチ
191  nn::atk::SoundSystem::AttachMemoryPool(&g_MemoryPool, nns::atk::GetPoolHeapAddress(), nns::atk::GetPoolHeapSize());
192 
193  // SoundArchivePlayer の初期化
194  std::size_t memSizeForSoundArchivePlayer = g_SoundArchivePlayer.GetRequiredMemSize( &g_SoundArchive );
195  g_pMemoryForSoundArchivePlayer = nns::atk::Allocate( memSizeForSoundArchivePlayer, nn::atk::SoundArchivePlayer::BufferAlignSize);
196  isSuccess = g_SoundArchivePlayer.Initialize(
197  &g_SoundArchive,
198  &g_SoundDataManager,
199  g_pMemoryForSoundArchivePlayer, memSizeForSoundArchivePlayer,
200  g_pMemoryForStreamBuffer, memSizeForStreamBuffer );
201  NN_ABORT_UNLESS( isSuccess, "cannot initialize SoundArchivePlayer" );
202 
203  // SoundActor の初期化
204  g_ActorPan = ActorPanDefault;
205  g_ActorVolume = ActorVolumeDefault;
206  g_SoundActor.Initialize( &g_SoundArchivePlayer );
207  g_SoundActor.SetPlayableSoundCount( 0, 2 );
208 }

275 bool UpdateAtk()
276 {
277  // StartSound / Stop
278  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::A >() )
279  {
280  PlayWithStartSound(SEQ_MARIOKART, "SEQ_MARIOKART");
281  }
282  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::X >() )
283  {
284  PlayWithStartSound(SE_YOSHI, "SE_YOSHI");
285  }
286  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Y >() )
287  {
288  PlayWithStartSound(STRM_MARIOKART, "STRM_MARIOKART");
289  }
290  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::B >() )
291  {
292  g_SoundActor.StopAllSound(3);
293  }
294 
295  // HoldSound
296  if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::Right >() )
297  {
298  g_ActorPan += ControlPadStep;
299  if (g_ActorPan > ActorPanMax)
300  {
301  g_ActorPan = ActorPanMax;
302  }
303  g_SoundActor.SetOutputPan(nn::atk::OutputDevice_Main, g_ActorPan);
304  PlayWithHoldSound(SE_WIHAHO, "SE_WIHAHO");
305  NNS_LOG("SetActorPan ... (%f)\n", g_ActorPan);
306  }
307  else if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::Up >() )
308  {
309  g_ActorVolume += ControlPadStep;
310  if (g_ActorVolume > ActorVolumeMax)
311  {
312  g_ActorVolume = ActorVolumeMax;
313  }
314  g_SoundActor.SetVolume(g_ActorVolume);
315  PlayWithHoldSound(SE_WIHAHO, "SE_WIHAHO");
316  NNS_LOG("SetActorVolume ... (%f)\n", g_ActorVolume);
317  }
318  else if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::Left >() )
319  {
320  g_ActorPan -= ControlPadStep;
321  if (g_ActorPan < ActorPanMin)
322  {
323  g_ActorPan = ActorPanMin;
324  }
325  g_SoundActor.SetOutputPan(nn::atk::OutputDevice_Main, g_ActorPan);
326  PlayWithHoldSound(SE_WIHAHO, "SE_WIHAHO");
327  NNS_LOG("SetActorPan ... (%f)\n", g_ActorPan);
328  }
329  else if ( nns::atk::IsHold< ::nn::hid::DebugPadButton::Down >() )
330  {
331  g_ActorVolume -= ControlPadStep;
332  if (g_ActorVolume < ActorVolumeMin)
333  {
334  g_ActorVolume = ActorVolumeMin;
335  }
336  g_SoundActor.SetVolume(g_ActorVolume);
337  PlayWithHoldSound(SE_WIHAHO, "SE_WIHAHO");
338  NNS_LOG("SetActorVolume ... (%f)\n", g_ActorVolume);
339  }
340 
341  // Print Usage
342  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::L >() )
343  {
344  PrintUsage();
345  }
346 
347  // Exit
348  if ( nns::atk::IsTrigger< ::nn::hid::DebugPadButton::Start >() )
349  {
350  return false;
351  }
352 
353  g_SoundArchivePlayer.Update();
354 
355  return true;
356 }

サンプルプログラムの解説

先のサンプルプログラムの Atk に関連している処理の流れは以下の通りです。(AtkSimple と違う部分には追加要素と書かれています)。

  • Atk ライブラリの初期化 (InitializeAtk())
    • AtkSimple で実行している初期化処理
    • SoundActor の初期化(追加要素)
  • SoundHeap にウェーブサウンドとシーケンスサウンドの再生に必要な情報をロード (LoadData())
  • メインループでの処理
    • キーによる処理
      • サウンドアクターでの再生・停止処理を実行(追加要素)
      • サウンドアクターの音量の増減(追加要素)
      • サウンドアクターのパンのかかり具合の変更(追加要素)
    • SoundArchivePlayer の更新
  • Atk ライブラリの終了処理 (FinalizeAtk())
    • SoundActor の終了処理(追加要素)

キー操作を行うことで、サウンドアクターのパラメータを変更する事ができます。 変更することのできるパラメータは以下になります。

  • 音量
  • パン

また、このサンプルでは取り扱っていませんが、サウンドアクターでは以下のパラメータも変更可能です。

  • ピッチ
  • LPF
  • BiquadFilter
  • 同時に再生可能なサウンド数
