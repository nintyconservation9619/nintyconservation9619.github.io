This tutorial shows how to load a Jpeg image from a file and compress it at runtime through a shader.

Overview

This tutorial shows how to load a Jpeg image from a file and compress it at runtime through a shader. The Jpeg data is used to create a linear NVN texture by running it through the Jpeg decoder and wrapping a memory pool around the resulting buffer. This texture is compressed into a DXT1 texture by running it through a shader containing the DXT1 compression algorithm. In order to render to the compressed texture, a texture view with a format that can be render to must be used. In the case DXT1 compression the texture format needs to be NVN_FORMAT_RGBA16UI.

456  * RuntimeTextureCompression::CompressJpegTexture
457  * ----------------------------------------------
458  * Compresses the Jpeg texture to the DXT1 format.
459  */
460 void RuntimeTextureCompression::CompressJpegTexture()
461 {
462  /*
463  * Setup a texture builder to create a DXT1 compressed
464  * texture. This texture will be rendered into using a
465  * specific NVN texture view and a shader to perform the
466  * compression.
467  */
468  NVNtextureBuilder textureBuilder;
469  nvnTextureBuilderSetDefaults(&textureBuilder);
470  nvnTextureBuilderSetDevice(&textureBuilder, &m_Device);
471  nvnTextureBuilderSetFormat(&textureBuilder, NVN_FORMAT_RGBA_DXT1);
472  nvnTextureBuilderSetTarget(&textureBuilder, NVN_TEXTURE_TARGET_2D);
473  nvnTextureBuilderSetSize2D(&textureBuilder, m_JpegWidth, m_JpegHeight);
474 
475  /* Get the size of the compressed texture. */
476  size_t compressedSize = nvnTextureBuilderGetStorageSize(&textureBuilder);
477 
478  /* Create a memory pool for the compressed texture. */
479  m_DXT1MemoryPool.Init(NULL, compressedSize, NVN_MEMORY_POOL_FLAGS_CPU_NO_ACCESS_BIT | NVN_MEMORY_POOL_FLAGS_GPU_CACHED_BIT, &m_Device);
480 
481  nvnTextureBuilderSetStorage(&textureBuilder, m_DXT1MemoryPool.GetMemoryPool(), 0);
482 
483  /* Initialize the tetxure. */
484  if (nvnTextureInitialize(&m_DXT1Texture, &textureBuilder) == NVN_FALSE)
485  {
486  NN_ASSERT(0, "Failed to initialize compressed texture.");
487  }
488 
489  /* Grab the texture ID. */
490  m_DXT1TextureID = m_pTextureIDManager->RegisterTexture(&m_JpegTexture);
491 
492  /* Create a sampler and get its ID. */
493  NVNsamplerBuilder samplerBuilder;
494  nvnSamplerBuilderSetDevice(&samplerBuilder, &m_Device);
495  nvnSamplerBuilderSetDefaults(&samplerBuilder);
496 
497  NVNboolean samplerSuccess = nvnSamplerInitialize(&m_DXT1Sampler, &samplerBuilder);
498  NN_ASSERT(samplerSuccess, "Failed to initialize sampler\n");
499  m_DXT1SamplerID = m_pTextureIDManager->RegisterSampler(&m_DXT1Sampler);
500 
501  /* Create the texture handle for the compressed texture. */
502  m_DXT1TextureHandle = nvnDeviceGetTextureHandle(&m_Device, m_DXT1TextureID, m_DXT1SamplerID);
503 
504  /*
505  * Create a texture view for the compressed texture. Using
506  * this texture view, the compressed texture can be set as
507  * a render target for the compression shader. The compression
508  * shader will read 4x4 blocks of pixels from the Jpeg texture,
509  * perform the compression algorithm, and write it out to the
510  * DXT1 texture through the view. For the DXT1 compression
511  * algorithm, the texture view need to specifically be of the
512  * type NVN_FORMAT_RGBA16UI.
513  */
514  nvnTextureViewSetDefaults(&m_DXT1CompressView);
515  nvnTextureViewSetFormat(&m_DXT1CompressView, NVN_FORMAT_RGBA16UI);
516 
517  /* Set default states for the draw. */
518  nvnDepthStencilStateSetDefaults(&m_DepthStencilState);
519  nvnPolygonStateSetDefaults(&m_PolygonState);
520  nvnMultisampleStateSetDefaults(&m_MultisampleState);
521  nvnColorStateSetDefaults(&m_ColorState);
522  nvnChannelMaskStateSetDefaults(&m_ChannelMaskState);
523  nvnBlendStateSetDefaults(&m_BlendState);
524 
525  /* Record the command buffer for the compression. */
526  m_pTextureCompressManagedCommandBuffer->BeginRecording();
527  {
528  NVNcommandBuffer* pCommandBuffer = m_pTextureCompressManagedCommandBuffer->GetCommandBuffer();
529 
530  /*
531  * Calculate the width and height of the compressed
532  * texture with the texture views format.
533  */
534  const int blockSize = 4;
535  int compressedWidth = (m_JpegWidth + blockSize - 1) / blockSize;
536  int compressedHeight = (m_JpegHeight + blockSize - 1) / blockSize;
537 
538  m_pTextureIDManager->SetSamplerPool(pCommandBuffer);
539  m_pTextureIDManager->SetTexturePool(pCommandBuffer);
540 
541  NVNtexture* renderTarget = &m_DXT1Texture;
542  NVNtextureView* targetView = &m_DXT1CompressView;
543 
544  /*
545  * Set the compressed texture as the render
546  * target along with the texture view.
547  */
548  nvnCommandBufferSetRenderTargets(pCommandBuffer, 1, &renderTarget, &targetView, NULL, NULL);
549  nvnCommandBufferSetViewport(pCommandBuffer, 0, 0, compressedWidth, compressedHeight);
550  nvnCommandBufferSetScissor(pCommandBuffer, 0, 0, compressedWidth, compressedHeight);
551 
552  /* Bind the shader program. */
553  std::vector<NVNProgramData*>& programData = m_pCompressShaderDataHolder->GetProgramData();
554  nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
555 
556  /* Bind the Jpeg texture to the proper shader stage/slot. */
557  nvnCommandBufferBindTexture(pCommandBuffer, NVN_SHADER_STAGE_FRAGMENT, 0, m_JpegTextureHandle);
558 
559  /* Bind the render states. */
560  nvnCommandBufferBindBlendState(pCommandBuffer, &m_BlendState);
561  nvnCommandBufferBindChannelMaskState(pCommandBuffer, &m_ChannelMaskState);
562  nvnCommandBufferBindColorState(pCommandBuffer, &m_ColorState);
563  nvnCommandBufferBindDepthStencilState(pCommandBuffer, &m_DepthStencilState);
564  nvnCommandBufferBindMultisampleState(pCommandBuffer, &m_MultisampleState);
565  nvnCommandBufferBindPolygonState(pCommandBuffer, &m_PolygonState);
566 
567  /*
568  * Draw a screen space quad to perform the compression
569  * algorithm with. No vertex buffer is bound as the vertex
570  * shader outputs positions based on the vertex ID alone.
571  */
572  nvnCommandBufferDrawArrays(pCommandBuffer, NVN_DRAW_PRIMITIVE_TRIANGLE_STRIP, 0, 4);
573 
574  /*
575  * Create a barrier in the command buffer to make sure that
576  * the texture is finished being compressed before being used.
577  */
578  nvnCommandBufferBarrier(pCommandBuffer, NVNbarrierBits::NVN_BARRIER_ORDER_FRAGMENTS_BIT | NVNbarrierBits::NVN_BARRIER_INVALIDATE_TEXTURE_BIT);
579  }
580  m_TextureCompressCommandHandle = m_pTextureCompressManagedCommandBuffer->EndRecording();
581 
582  /* Submit the command buffer for the compression. */
583  nvnQueueSubmitCommands(&m_Queue, 1, &m_TextureCompressCommandHandle);
584 }

1057  * ---------------------------------
1058  * This method is called everytime the window is resized and is passed
1059  * the new size of the window. It frees the old render target and creates a new
1060  * render target with the new screen size. A new set of commands for the command buffer
1061  * is recorded.
1062  */
1063 void RuntimeTextureCompression::Resize(int width, int height)
1064 {
1065  m_ScreenWidth = width;
1066  m_ScreenHeight = height;
1067 
1068  /* Check for the window being minimized or having no visible surface. */
1069  if (width == 0 || height == 0)
1070  {
1071  return;
1072  }
1073 
1074  /* If it's the first time Resize is called, allocate the NVNwindow. */
1075  if (m_pWindow == NULL)
1076  {
1077  m_pWindow = new NVNwindow;
1078  }
1079  /*
1080  * Otherwise finalize (free) the NVNwindow used for the previous window size.
1081  * The NVNWindow must be finalized before a render target it owns is finalized.
1082  */
1083  else
1084  {
1085  nvnWindowFinalize(m_pWindow);
1086  }
1087 
1088  /* Set up the builder for the render target. */
1089  nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
1090  nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_DISPLAY_BIT | NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
1091  nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
1092  nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
1093  nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_RGBA8);
1094 
1095  for(int i = 0; i < g_NumColorBuffers; ++i)
1096  {
1097  /* If it's the first time Resize is called, allocate the render target. */
1098  if(m_RenderTargets[i] == NULL)
1099  {
1100  m_RenderTargets[i] = new NVNtexture;
1101  }
1102  /* Otherwise finalize (free) the render target used for the previous window size. */
1103  else
1104  {
1105  nvnTextureFinalize(m_RenderTargets[i]);
1106  }
1107 
1108  nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * i);
1109 
1110  /* Create the texture using the current state of the texture builder. */
1111  nvnTextureInitialize(m_RenderTargets[i], &m_RenderTargetBuilder);
1112  }
1113 
1114  if (m_pDepthTarget == NULL)
1115  {
1116  m_pDepthTarget = new NVNtexture;
1117  }
1118  else
1119  {
1120  nvnTextureFinalize(m_pDepthTarget);
1121  }
1122 
1123  /* Initialize depth buffer for render target. */
1124  nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
1125  nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
1126  nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
1127  nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
1128  nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_DEPTH32F);
1129  nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * g_NumColorBuffers);
1130  nvnTextureInitialize(m_pDepthTarget, &m_RenderTargetBuilder);
1131 
1132  /* Pass off the render targets to the window. */
1133  nvnWindowBuilderSetTextures(&m_WindowBuilder, g_NumColorBuffers, m_RenderTargets);
1134  nvnWindowInitialize(m_pWindow, &m_WindowBuilder);
1135 
1136  /* Remake the projection matrix with the new screen size. */
1137  nn::util::MatrixPerspectiveFieldOfViewRightHanded(&m_ProjMtx, 60.0f * 3.14159f / 180.0f, static_cast<float>(m_ScreenWidth) / static_cast<float>(m_ScreenHeight), 1.0f, 1000.0f);
1138 }
